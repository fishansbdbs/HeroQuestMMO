  const weapon = getItem(game.save.equippedWeapon);
  const cooldown = pulse.cooldownMs * (weapon?.pulseCooldownBonus || 1);
  const left = Math.max(0, cooldown - (now - game.cooldowns.ability));
  ui.abilityText.textContent = left > 0 ? `${Math.ceil(left / 1000)}s` : "Pulse";
  updatePrompt();
  updateQuestTracker();
  updateBossHud();
  updateNetworkStatus();
  updateFps(now);
}

function updateQuestTracker() {
  ui.questTracker.innerHTML = getQuestList().slice(0, 5).map((quest) => {
    const progress = game.save.questProgress[quest.id] || { current: 0, complete: false };
    return `<div class="quest-item ${progress.complete ? "complete" : ""}">${quest.name}: ${progress.current}/${quest.required}</div>`;
  }).join("");
}

function updateBossHud() {
  const active = game.zone === ZONES.BOSS && game.boss && !game.boss.defeated;
  ui.bossFrame.classList.toggle("active", active);
  if (!active) return;
  ui.bossBar.style.width = `${(game.boss.health / game.boss.maxHealth) * 100}%`;
  ui.bossPhase.textContent = `Phase ${game.boss.phase}`;
  ui.bossWarning.textContent = game.boss.attack ? bossAttackText(game.boss.attack.type) : "Watch the wings and tail.";
}

function updateNetworkStatus() {
  if (!ui.networkStatus) return;
  ui.networkStatus.textContent = game.online ? "Online" : "Solo local";
  ui.networkStatus.className = game.online ? "online" : "offline";
}

function updateFps(now) {
  game.fps.frames += 1;
  if (now - game.fps.last >= 1000) {
    game.fps.value = game.fps.frames;
    game.fps.frames = 0;
    game.fps.last = now;
  }
  if (game.save.settings.showFps) {
    ui.zoneLabel.textContent = `${getZone(game.zone).name} - ${game.fps.value} FPS`;
  }
}

function updatePrompt() {
  game.promptAction = null;
  const player = groupPosition(game.playerGroup);
  for (const portal of game.portals) {
    if (distance2d(player, portal) <= portal.radius) {
      const locked = game.save.level < portal.minLevel;
      game.promptAction = () => {
        if (locked) {
          toast(`Reach level ${portal.minLevel} to enter ${portal.label}.`);
          return;
        }
        loadZone(portal.targetZone, true);
      };
      ui.prompt.textContent = locked ? `Locked: ${portal.label} requires level ${portal.minLevel}` : `E: Travel to ${portal.label}`;
      ui.prompt.classList.remove("hidden");
      return;
    }
  }
  const nearestLoot = findNearestLoot(3.2);
  if (nearestLoot) {
    game.promptAction = () => claimLoot(nearestLoot.id);
    ui.prompt.textContent = `E: Collect ${lootLabel(nearestLoot)}`;
    ui.prompt.classList.remove("hidden");
    return;
  }
  if (game.zone === ZONES.HUB && distance2d(player, { x: 8, z: 16 }) < 4) {
    game.promptAction = () => openPanel("quests");
    ui.prompt.textContent = "E: Read Quest Board";
    ui.prompt.classList.remove("hidden");
    return;
  }
  if (game.zone === ZONES.HUB && distance2d(player, { x: -10, z: 16 }) < 4) {
    game.promptAction = () => openPanel("inventory");
    ui.prompt.textContent = "E: Browse Equipment";
    ui.prompt.classList.remove("hidden");
    return;
  }
  ui.prompt.classList.add("hidden");
}

function interact() {
  if (game.promptAction) {
    playUiSound("tick");
    game.promptAction();
  }
}

function basicAttack() {
  const now = performance.now();
  if (now - game.cooldowns.attack < PLAYER_LIMITS.attackCooldownMs) return;
  game.cooldowns.attack = now;
  playUiSound("swing");
  swingWeapon();
  if (game.zone === ZONES.BOSS && game.boss && !game.boss.defeated && distance2d(groupPosition(game.playerGroup), { x: 0, z: 0 }) < 8) {
    hitBoss(calculatePlayerDamage(game.save));
    if (game.online) game.socket.emit(NET.COMBAT_ATTACK, { targetId: "shadow_wyrm" }, handleCombatAck);
    return;
  }
  const target = findNearestEnemy(PLAYER_LIMITS.meleeRange);
  if (!target) return;
  if (game.online) {
    game.socket.emit(NET.COMBAT_ATTACK, { targetId: target.id }, handleCombatAck);
  } else {
    damageLocalEnemy(target.id, calculatePlayerDamage(game.save));
  }
}

function useHeroPulse() {
  const now = performance.now();
  const weapon = getItem(game.save.equippedWeapon);
  const cooldown = ABILITIES.hero_pulse.cooldownMs * (weapon?.pulseCooldownBonus || 1);
  if (now - game.cooldowns.ability < cooldown) return;
  game.cooldowns.ability = now;
  playUiSound("pulse");
  pulseEffect();
  if (game.online) {
    game.socket.emit(NET.COMBAT_ABILITY, {}, handleCombatAck);
    return;
  }
  for (const enemy of game.enemies.values()) {
    if (distance2d(enemy.position, groupPosition(game.playerGroup)) <= ABILITIES.hero_pulse.radius) {
      damageLocalEnemy(enemy.id, Math.round(calculatePlayerDamage(game.save) * ABILITIES.hero_pulse.damageScale));
    }
  }
  if (game.zone === ZONES.BOSS && game.boss && distance2d(groupPosition(game.playerGroup), { x: 0, z: 0 }) <= 8) {
    hitBoss(Math.round(calculatePlayerDamage(game.save) * ABILITIES.hero_pulse.damageScale));
  }
}

function dash() {
  const now = performance.now();
  if (now - game.cooldowns.dash < PLAYER_LIMITS.dashCooldownMs) return;
  game.cooldowns.dash = now;
  const forward = new THREE.Vector3(Math.sin(game.playerGroup.rotation.y), 0, Math.cos(game.playerGroup.rotation.y));
  game.playerGroup.position.addScaledVector(forward, 4);
  clampToZone(game.playerGroup.position);
  playUiSound("dash");
}

function usePotion() {
  const result = consumeInventoryItem(game.save.inventory, "small_health_potion", 1);
  if (!result.consumed) {
    toast("No health potion available.");
    return;
  }
  game.save.inventory = result.inventory;
  game.save.health = Math.min(game.save.maxHealth, game.save.health + ITEMS.small_health_potion.heal);
  playUiSound("pickup");
  toast("Health potion used.");
  saveGame();
}

function damageLocalEnemy(enemyId, damage) {
  const enemy = game.enemies.get(enemyId);
  if (!enemy) return;
  enemy.health = Math.max(0, enemy.health - damage);
  floatingText(`-${damage}`, enemy.position, "#ffdd9a");
  playUiSound("hit");
  if (enemy.health <= 0) {
    const def = getEnemy(enemy.enemyId);
    awardEnemy(def);
    spawnLocalLoot(def, enemy.position);
    game.enemies.delete(enemyId);
    const mesh = game.enemyMeshes.get(enemyId);
    if (mesh) {
      game.scene.remove(mesh);
      game.enemyMeshes.delete(enemyId);
    }
  }
}

function awardEnemy(enemyDef) {
  const loot = rollLoot(enemyDef);
  const beforeLevel = game.save.level;
  game.save = addProgressRewards(game.save, { xp: loot.xp, coins: loot.coins });
  const questUpdate = applyQuestKill(game.save.questProgress, enemyDef);
  game.save.questProgress = questUpdate.progress;
  if (game.save.level > beforeLevel) {
    toast(`Level ${game.save.level}!`);
    playUiSound("level");
  }
  for (const quest of questUpdate.completed) {
    toast(`Quest complete: ${quest.name}`);
    game.save = addProgressRewards(game.save, quest.reward);
    if (quest.reward.itemId) game.save.inventory = addInventoryItem(game.save.inventory, quest.reward.itemId, 1);
    if (quest.reward.title) game.save.title = quest.reward.title;
  }
  saveGame();
}

function spawnLocalLoot(enemyDef, position) {
  const drop = rollLoot(enemyDef);
  const id = `local_loot_${game.localLootId++}`;
  game.loot.set(id, {
    id,
    zone: game.zone,
    position: { ...position, y: 0.25 },
    coins: 0,
    items: drop.items
  });
}

function claimLoot(lootId) {
  if (game.online) {
    game.socket.emit(NET.LOOT_CLAIM, { lootId }, (ack) => {
      if (ack?.ok) {
        applyLootBag(ack.bag);
        game.loot.delete(lootId);
        toast("Loot collected.");
      }
    });
    return;
  }
  const bag = game.loot.get(lootId);
  if (!bag) return;
  applyLootBag(bag);
  game.loot.delete(lootId);
  const mesh = game.lootMeshes.get(lootId);
  if (mesh) {
    game.scene.remove(mesh);
    game.lootMeshes.delete(lootId);
  }
  toast("Loot collected.");
}

function applyLootBag(bag) {
  game.save.coins += bag.coins || 0;
  for (const item of bag.items || []) {
    game.save.inventory = addInventoryItem(game.save.inventory, item.itemId, item.quantity || 1);
    toast(`Found ${getItem(item.itemId)?.name || item.itemId}.`);
  }
  playUiSound("pickup");
  saveGame();
}

function hitBoss(damage) {
  if (!game.boss || game.boss.defeated) return;
  game.boss.health = Math.max(0, game.boss.health - damage);
  game.boss.phase = game.boss.health <= game.boss.maxHealth * 0.5 ? 2 : 1;
  floatingText(`-${damage}`, { x: 0, y: 4, z: -1 }, "#ff8d76");
  if (game.boss.health <= 0) defeatBoss();
}

function defeatBoss() {
  game.boss.defeated = true;
  clearTelegraphs();
  const beforeLevel = game.save.level;
  game.save = addProgressRewards(game.save, { xp: BOSS.xp, coins: BOSS.coins });
  game.save.title = "Wyrm-Touched";
  const questUpdate = applyQuestKill(game.save.questProgress, { id: BOSS.id, family: BOSS.family });
  game.save.questProgress = questUpdate.progress;
  const drop = rollLoot(BOSS);
  const lootId = `boss_loot_${game.localLootId++}`;
  game.loot.set(lootId, { id: lootId, zone: ZONES.BOSS, position: { x: 0, y: 0.25, z: 0 }, coins: 0, items: drop.items });
  toast("Shadow Wyrm defeated!");
  if (game.save.level > beforeLevel) toast(`Level ${game.save.level}!`);
  playUiSound("boss");
  saveGame();
}

function handleCombatAck(result) {
  if (!result?.ok) return;
  if (result.result?.defeated) {
    awardEnemy(result.result.enemyDef);
    if (result.result.lootBag) game.loot.set(result.result.lootBag.id, result.result.lootBag);
  }
  const bossResult = result.boss || result.result?.boss;
  if (bossResult?.boss) {
    game.boss = { ...game.boss, ...bossResult.boss };
  }
  if (bossResult?.defeated) {
    game.save.title = "Wyrm-Touched";
    if (bossResult.lootBag) game.loot.set(bossResult.lootBag.id, bossResult.lootBag);
    toast("Shadow Wyrm defeated!");
  }
}

function damagePlayer(amount) {
  game.save.health = Math.max(0, game.save.health - amount);
  floatingText(`-${amount}`, groupPosition(game.playerGroup), "#ff8080");
  playUiSound("hurt");
  if (game.save.health <= 0) {
    toast("You were knocked back to Dawnrest.");
    game.save.health = game.save.maxHealth;
    loadZone(ZONES.HUB, true);
  }
  saveGame();
}

function spawnLocalEnemies() {
  game.enemies.clear();
  for (const spawn of FIELD_SPAWNS) {
    for (let i = 0; i < spawn.count; i += 1) {
      const angle = Math.random() * Math.PI * 2;
      const radius = Math.random() * spawn.radius;
      const def = getEnemy(spawn.enemyId);
      const id = `local_enemy_${game.localEnemyId++}`;
      game.enemies.set(id, {
        id,
        enemyId: spawn.enemyId,
        health: def.maxHealth,
        maxHealth: def.maxHealth,
        position: {
          x: spawn.center[0] + Math.cos(angle) * radius,
          y: 0,
          z: spawn.center[2] + Math.sin(angle) * radius
        }
      });
    }
  }
}

function startLocalBoss() {
  game.boss = {
    id: BOSS.id,
    name: BOSS.name,
    health: BOSS.maxHealth,
    maxHealth: BOSS.maxHealth,
    phase: 1,
    active: true,
    defeated: false,
    attack: null,
    nextAttackAt: performance.now() + 3000
  };
}

function resolveBossAttack(type) {
  if (type === "shadow_summon") {
    for (let i = 0; i < 3; i += 1) {
      const def = ENEMIES.shadow_slime;
      const id = `shadow_${game.localEnemyId++}`;
      const a = Math.random() * Math.PI * 2;
      game.enemies.set(id, {
        id,
        enemyId: def.id,
        health: def.maxHealth,
        maxHealth: def.maxHealth,
        position: { x: Math.cos(a) * 8, y: 0, z: Math.sin(a) * 8 }
      });
    }
    toast("The Shadow Wyrm summons slimes.");
    return;
  }
  const pos = groupPosition(game.playerGroup);
  const dist = distance2d(pos, { x: 0, z: 0 });
  const hit = type === "fire_cone" ? pos.z > -7 && Math.abs(pos.x) < 10 && dist < 20 : dist < 12;
  if (hit) damagePlayer(calculateIncomingDamage(BOSS.damage, game.save));
  if (type === "wing_gust") {
    const push = new THREE.Vector3(pos.x, 0, pos.z).normalize().multiplyScalar(4);
    game.playerGroup.position.add(push);
  }
}

function showTelegraph(type) {
  clearTelegraphs();
  let mesh;
  if (type === "fire_cone") {
    const shape = new THREE.Shape();
    shape.moveTo(0, 0);
    shape.lineTo(-8, 17);
    shape.lineTo(8, 17);
    shape.lineTo(0, 0);
    mesh = new THREE.Mesh(new THREE.ShapeGeometry(shape), mat(0xff5533, 0.33, true));
    mesh.rotation.x = -Math.PI / 2;
    mesh.position.set(0, 0.12, -4);
  } else {
    mesh = new THREE.Mesh(new THREE.RingGeometry(type === "tail_sweep" ? 7 : 11, type === "tail_sweep" ? 12 : 15, 48), mat(type === "tail_sweep" ? 0xff8b3d : 0x9ad5ff, 0.3, true));
    mesh.rotation.x = -Math.PI / 2;
    mesh.position.y = 0.13;
  }
  game.scene.add(mesh);
  game.telegraphs.push(mesh);
}

function clearTelegraphs() {
  game.telegraphs.forEach((mesh) => game.scene.remove(mesh));
  game.telegraphs = [];
}

function applySnapshot(snapshot) {
  if (!snapshot || snapshot.zone !== game.zone) return;
  if (snapshot.enemies) {
    game.enemies.clear();
    snapshot.enemies.forEach((enemy) => game.enemies.set(enemy.id, enemy));
  }
  if (snapshot.loot) {
    game.loot.clear();
    snapshot.loot.forEach((bag) => game.loot.set(bag.id, bag));
  }
  if (snapshot.boss) {
    game.boss = { ...game.boss, ...snapshot.boss };
  }
  syncRemotePlayers(snapshot.players || []);
  updatePartyList(snapshot.parties || [], snapshot.players || []);
}

function syncRemotePlayers(players) {
  const seen = new Set();
  for (const player of players) {
    if (player.id === game.socket?.id) {
      game.save.health = player.health ?? game.save.health;
      continue;
    }
    seen.add(player.id);
    let remote = game.remotePlayers.get(player.id);
    if (!remote) {
      const mesh = createPlayerMesh(player.name, player.color, false);
      game.scene.add(mesh);
      remote = { mesh, target: null };
      game.remotePlayers.set(player.id, remote);
    }
    remote.target = new THREE.Vector3(player.position.x, 0, player.position.z);
    remote.mesh.visible = player.zone === game.zone;
  }
  for (const [id, remote] of game.remotePlayers.entries()) {
    if (!seen.has(id)) {
      game.scene.remove(remote.mesh);
      game.remotePlayers.delete(id);
    }
  }
}

function updatePartyList(parties, players) {
  const myParty = parties.find((party) => party.members.includes(game.socket?.id));
  if (!myParty) {
    ui.partyList.textContent = "No party";
    return;
  }
  const names = myParty.members.map((id) => players.find((player) => player.id === id)?.name || "Hero");
  ui.partyList.innerHTML = `<strong>Party ${myParty.code}</strong><br>${names.join("<br>")}`;
}

function syncEnemyMeshes() {
  for (const [id, enemy] of game.enemies.entries()) {
    if (!game.enemyMeshes.has(id)) {
      const mesh = createEnemyMesh(enemy);
      game.scene.add(mesh);
      game.enemyMeshes.set(id, mesh);
    }
  }
  for (const [id, mesh] of game.enemyMeshes.entries()) {
    if (!game.enemies.has(id)) {
      game.scene.remove(mesh);
      game.enemyMeshes.delete(id);
    }
  }
}

function syncLootMeshes() {
  for (const [id, bag] of game.loot.entries()) {
    if (!game.lootMeshes.has(id)) {
      const mesh = createLootMesh(bag);
      game.scene.add(mesh);
      game.lootMeshes.set(id, mesh);
    }
