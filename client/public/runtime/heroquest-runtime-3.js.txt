  updateCombatLog(now);
  updatePrompt();
  updateQuestTracker();
  updateBossHud();
  updateNetworkStatus();
  updateFps(now);
}

function updateQuestTracker() {
  if (game.questCollapsed) {
    ui.questTracker.innerHTML = "";
    return;
  }
  const frostQuestIds = new Set(["frozen_road", "cold_blooded", "howling_white", "shattered_ward", "heart_of_the_blizzard", "palace_of_zero"]);
  const palaceQuestIds = new Set(["palace_of_zero", "icezero"]);
  const quests = game.zone === ZONES.PALACE
    ? getQuestList().filter((quest) => palaceQuestIds.has(quest.id))
    : game.zone === ZONES.FROSTVEIL
      ? getQuestList().filter((quest) => frostQuestIds.has(quest.id))
    : getQuestList().slice(0, 5);
  ui.questTracker.innerHTML = quests.map((quest) => {
    const progress = game.save.questProgress[quest.id] || { current: 0, complete: false };
    return `<div class="quest-item ${progress.complete ? "complete" : ""}"><strong>${quest.name}</strong><span>${progress.current}/${quest.required}</span></div>`;
  }).join("");
}

function updateBossHud() {
  const activeShadow = game.zone === ZONES.BOSS && game.boss && !game.boss.defeated;
  const activeIceMage = game.zone === ZONES.PALACE && game.iceMage && !game.iceMage.defeated;
  const active = activeShadow || activeIceMage;
  ui.bossFrame.classList.toggle("active", active);
  if (!active) return;
  const bossState = activeIceMage ? game.iceMage : game.boss;
  if (ui.bossName) ui.bossName.textContent = activeIceMage ? ICE_MAGE_BOSS.name : BOSS.name;
  ui.bossBar.style.width = `${(bossState.health / bossState.maxHealth) * 100}%`;
  ui.bossPhase.textContent = `Phase ${bossState.phase}`;
  ui.bossWarning.textContent = bossState.attack ? bossAttackText(bossState.attack.type) : activeIceMage ? "Watch the ice floor." : "Watch the wings and tail.";
}

function updateNetworkStatus() {
  if (!ui.networkStatus) return;
  ui.networkStatus.textContent = game.online ? "Online" : "Solo local";
  ui.networkStatus.className = game.online ? "online" : "offline";
  if (ui.pingText) ui.pingText.textContent = game.online && game.lastPingMs != null ? `${game.lastPingMs} ms` : "-- ms";
}

function updateTargetFrame() {
  const target = getCurrentTarget();
  if (target?.id === "shadow_wyrm" || target?.id === ICE_MAGE_BOSS.id) {
    ui.targetFrame.classList.add("hidden");
    return;
  }
  ui.targetFrame.classList.toggle("hidden", !target);
  if (!target) return;
  const pct = Math.max(0, Math.min(100, (target.health / Math.max(1, target.maxHealth)) * 100));
  ui.targetName.textContent = targetDisplayName(target);
  ui.targetMeta.textContent = `Lv. ${target.level || "?"}`;
  ui.targetBar.style.width = `${pct}%`;
  ui.targetHealth.textContent = `${Math.ceil(target.health)} / ${target.maxHealth} HP`;
  ui.targetMode.textContent = game.autoAttack ? "Auto attacking" : "Target locked";
}

function updateHotbar(now) {
  ui.hotbar?.querySelectorAll("[data-hotbar-action]").forEach((button) => {
    const baseSlot = HOTBAR_DEFS.find((entry) => entry.slot === Number(button.dataset.hotbarSlot));
    const slot = resolveHotbarSlot(baseSlot);
    const cooldown = cooldownLeft(slot, now);
    const notLearned = Boolean(slot?.requiresAbility && !knowsAbility(slot.requiresAbility));
    const overlay = button.querySelector(".slot-cooldown");
    const count = button.querySelector(".slot-count");
    button.dataset.hotbarAction = slot.action;
    button.querySelector(".slot-key").textContent = slot.key;
    button.querySelector(".slot-icon").textContent = slot.icon;
    button.querySelector(".slot-name").textContent = slot.name;
    button.classList.toggle("cooling", cooldown > 0);
    button.classList.toggle("active", slot.action === "auto" && game.autoAttack);
    button.classList.toggle("pressed", game.hotbarPressed === slot.action);
    button.classList.toggle("locked", Boolean(slot.locked) || notLearned);
    button.title = notLearned ? `Learn ${slot.name} before using it.` : slot.tooltip;
    button.disabled = isLocallyDead() || Boolean(slot.locked) || notLearned || (slot.action !== "auto" && cooldown > 0) || (slot.action === "potion" && countItem("small_health_potion") <= 0);
    if (overlay) overlay.style.height = cooldown > 0 ? `${Math.min(100, (cooldown / Math.max(1, slot.cooldownMs)) * 100)}%` : "0%";
    if (count) {
      if (slot.action === "potion") count.textContent = `x${countItem("small_health_potion")}`;
      else if (ABILITIES[slot.action]?.manaCost && !notLearned) count.textContent = game.save.mana < ABILITIES[slot.action].manaCost ? "Mana" : "";
      else if (cooldown > 0) count.textContent = `${Math.ceil(cooldown / 1000)}`;
      else count.textContent = "";
    }
  });
}

function knowsAbility(abilityId) {
  return (game.save.learnedAbilities || []).includes(abilityId);
}

function cooldownLeft(slot, now) {
  if (!slot || slot.locked || !slot.cooldownMs) return 0;
  const key = slot.action === "auto" ? "attack" : slot.action;
  if (slot.action === "hero_pulse") {
    const weapon = getItem(game.save.equippedWeapon);
    return game.cooldowns.pulse ? Math.max(0, ABILITIES.hero_pulse.cooldownMs * (weapon?.pulseCooldownBonus || 1) - (now - game.cooldowns.pulse)) : 0;
  }
  return game.cooldowns[key] ? Math.max(0, slot.cooldownMs - (now - game.cooldowns[key])) : 0;
}

function updateCombatLog(now) {
  game.combatLog = game.combatLog.filter((entry) => now - entry.at < 5200).slice(-5);
  if (!ui.combatLog) return;
  ui.combatLog.innerHTML = game.combatLog.map((entry) => `<div>${escapeHtml(entry.text)}</div>`).join("");
}

function logCombat(text) {
  game.combatLog.push({ text, at: performance.now() });
  if (game.combatLog.length > 8) game.combatLog.shift();
}

function selectTarget(target) {
  game.targetId = typeof target === "string" ? target : target?.id || null;
  if (!game.targetId) game.autoAttack = false;
  updateTargetFrame();
  updateTargetRing();
}

function getCurrentTarget() {
  if (game.targetId === "shadow_wyrm" && game.zone === ZONES.BOSS && game.boss && !game.boss.defeated) {
    return { id: "shadow_wyrm", name: BOSS.name, level: BOSS.level, health: game.boss.health, maxHealth: game.boss.maxHealth, position: { x: 0, y: 0, z: 0 } };
  }
  if (game.targetId === ICE_MAGE_BOSS.id && game.zone === ZONES.PALACE && game.iceMage && !game.iceMage.defeated) {
    return { id: ICE_MAGE_BOSS.id, name: ICE_MAGE_BOSS.name, level: ICE_MAGE_BOSS.level, health: game.iceMage.health, maxHealth: game.iceMage.maxHealth, position: { x: 0, y: 0, z: 0 } };
  }
  const enemy = game.targetId ? game.enemies.get(game.targetId) : null;
  if (!enemy || enemy.health <= 0) {
    game.targetId = null;
    game.autoAttack = false;
    return null;
  }
  const def = getEnemy(enemy.enemyId);
  return { ...enemy, name: def.name, level: def.level, position: enemy.position };
}

function targetDisplayName(target) {
  if (!target) return "No target";
  if (target.id === "shadow_wyrm") return BOSS.name;
  if (target.id === ICE_MAGE_BOSS.id) return ICE_MAGE_BOSS.name;
  return target.name || getEnemy(target.enemyId)?.name || "Enemy";
}

function updateTargetRing() {
  const target = getCurrentTarget();
  if (!target) {
    if (game.targetRing) game.targetRing.visible = false;
    return;
  }
  if (!game.targetRing) {
    const ring = new THREE.Mesh(new THREE.RingGeometry(0.85, 1.1, 48), mat(0xffe38b, 0.72, true));
    ring.rotation.x = -Math.PI / 2;
    ring.position.y = 0.08;
    game.scene.add(ring);
    game.targetRing = ring;
  }
  game.targetRing.visible = true;
  game.targetRing.position.set(target.position.x, 0.08, target.position.z);
  const scale = target.id === "shadow_wyrm" ? 3.2 : target.id === ICE_MAGE_BOSS.id ? 2.15 : 1;
  game.targetRing.scale.setScalar(scale + Math.sin(performance.now() * 0.006) * 0.05);
}

function pickEnemyFromPointer(event) {
  if (!game.camera) return null;
  const pointer = new THREE.Vector2((event.clientX / window.innerWidth) * 2 - 1, -(event.clientY / window.innerHeight) * 2 + 1);
  const raycaster = new THREE.Raycaster();
  raycaster.setFromCamera(pointer, game.camera);
  const pickables = [];
  for (const mesh of game.enemyMeshes.values()) mesh.traverse((child) => { if (child.isMesh) pickables.push(child); });
  if (game.bossGroup && game.zone === ZONES.BOSS && game.boss && !game.boss.defeated) {
    game.bossGroup.traverse((child) => {
      if (child.isMesh) {
        child.userData.enemyInstanceId = "shadow_wyrm";
        pickables.push(child);
      }
    });
  }
  if (game.iceMageGroup && game.zone === ZONES.PALACE && game.iceMage && !game.iceMage.defeated) {
    game.iceMageGroup.traverse((child) => {
      if (child.isMesh) {
        child.userData.enemyInstanceId = ICE_MAGE_BOSS.id;
        pickables.push(child);
      }
    });
  }
  const hit = raycaster.intersectObjects(pickables, true)[0];
  let object = hit?.object || null;
  while (object) {
    if (object.userData.enemyInstanceId === "shadow_wyrm") return "shadow_wyrm";
    if (object.userData.enemyInstanceId === ICE_MAGE_BOSS.id) return ICE_MAGE_BOSS.id;
    if (object.userData.enemyInstanceId) return object.userData.enemyInstanceId;
    object = object.parent;
  }
  return null;
}

function updateFps(now) {
  game.fps.frames += 1;
  if (now - game.fps.last >= 1000) {
    game.fps.value = game.fps.frames;
    game.fps.frames = 0;
    game.fps.last = now;
  }
  if (game.save.settings.showFps) {
    ui.zoneLabel.textContent = `${getZone(game.zone).name} - ${game.fps.value} FPS`;
  }
}

function updatePrompt() {
  game.promptAction = null;
  if (isLocallyDead()) {
    ui.prompt.classList.add("hidden");
    return;
  }
  const player = groupPosition(game.playerGroup);
  const nearestNpc = findNearestNpc(3.2);
  if (nearestNpc) {
    game.promptAction = () => openDialogue(nearestNpc.dialogueId);
    ui.prompt.textContent = `E: Talk to ${nearestNpc.label}`;
    ui.prompt.classList.remove("hidden");
    return;
  }
  for (const portal of game.portals) {
    if (distance2d(player, portal) <= portal.radius) {
      const gate = canEnterZone(game.save, portal.targetZone);
      const locked = !gate.ok || game.save.level < portal.minLevel;
      game.promptAction = () => {
        if (locked) {
          toast(portal.lockMessage || gate.message || `Reach level ${portal.minLevel} to enter ${portal.label}.`);
          return;
        }
        loadZone(portal.targetZone, true);
      };
      ui.prompt.textContent = locked ? `Locked: ${portal.label} requires level ${portal.minLevel}` : `E: Travel to ${portal.label}`;
      ui.prompt.classList.remove("hidden");
      return;
    }
  }
  const nearestLoot = findNearestLoot(3.2);
  if (nearestLoot) {
    game.promptAction = () => claimLoot(nearestLoot.id);
    ui.prompt.textContent = `E: Collect ${lootLabel(nearestLoot)}`;
    ui.prompt.classList.remove("hidden");
    return;
  }
  const nearestChest = findNearestChest(3.4);
  if (nearestChest) {
    game.promptAction = () => openChest(nearestChest.id);
    ui.prompt.textContent = nearestChest.opened ? "Treasure chest is empty" : "E: Open Treasure Chest";
    ui.prompt.classList.remove("hidden");
    return;
  }
  if (game.zone === ZONES.FROSTVEIL && distance2d(player, { x: 8, z: -8 }) < 5) {
    const event = game.publicEvent || {};
    if (event.phase === "starting") {
      ui.prompt.textContent = "Frost Ward defense is starting.";
    } else if (event.active) {
      ui.prompt.textContent = `Frost Ward wave ${event.wave || 1}/${event.totalWaves || 3}`;
    } else if (event.phase === "cooldown") {
      ui.prompt.textContent = "Frost Ward is recharging.";
    } else {
      game.promptAction = () => activatePublicEvent("defend_frost_ward");
      ui.prompt.textContent = "E: Activate Frost Ward";
    }
    ui.prompt.classList.remove("hidden");
    return;
  }
  if (game.zone === ZONES.HUB && distance2d(player, { x: 16, z: -12 }) < 4) {
    game.promptAction = () => {
      selectTarget("training_dummy");
      toast("Training dummy targeted.");
    };
    ui.prompt.textContent = "E: Target Training Dummy";
    ui.prompt.classList.remove("hidden");
    return;
  }
  if (game.zone === ZONES.HUB && distance2d(player, { x: 8, z: 16 }) < 4) {
    game.promptAction = () => openPanel("quests");
    ui.prompt.textContent = "E: Read Quest Board";
    ui.prompt.classList.remove("hidden");
    return;
  }
  if (game.zone === ZONES.HUB && distance2d(player, { x: -10, z: 16 }) < 4) {
    game.promptAction = () => openPanel("shop");
    ui.prompt.textContent = "E: Browse Weapon Shop";
    ui.prompt.classList.remove("hidden");
    return;
  }
  const breadcrumb = currentBreadcrumb();
  if (breadcrumb) {
    ui.prompt.textContent = breadcrumb;
    ui.prompt.classList.remove("hidden");
    return;
  }
  ui.prompt.classList.add("hidden");
}

function currentBreadcrumb() {
  if (game.zone === ZONES.HUB && game.save.level >= 2 && !knowsAbility("hero_pulse")) {
    return "Marker: Arcanist Mira trains magic near the plaza.";
  }
  if (game.zone === ZONES.HUB && game.save.level >= 5 && !(game.save.waypoints || []).includes("frostveil_camp")) {
    return "Marker: Greenvale portal leads toward Frostveil.";
  }
  if (game.zone === ZONES.FIELD && game.save.level >= 5 && !game.save.questProgress?.frozen_road?.complete) {
    return "Marker: Frostveil portal waits on the northwest ridge.";
  }
  if (game.zone === ZONES.FROSTVEIL && !game.save.questProgress?.shattered_ward?.complete) {
    return "Marker: Frozen wards glow across Frostveil.";
  }
  if (game.zone === ZONES.FROSTVEIL && game.save.level >= 8 && !game.save.questProgress?.palace_of_zero?.complete) {
    return "Marker: Palace gate rises northeast.";
  }
  if (game.zone === ZONES.PALACE && !game.save.questProgress?.icezero?.complete) {
    return "Marker: Zero waits in the central chamber.";
  }
  return "";
}

function interact() {
  if (isLocallyDead()) {
    toast("You cannot interact while defeated.");
    return;
  }
  if (game.promptAction) {
