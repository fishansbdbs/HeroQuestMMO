  }
  for (const [id, mesh] of game.lootMeshes.entries()) {
    if (!game.loot.has(id)) {
      game.scene.remove(mesh);
      game.lootMeshes.delete(id);
    }
  }
}

function createLootMesh(bag) {
  const group = new THREE.Group();
  const rare = bag.items?.some((item) => ["shadow_fang", "wyrm_scale_shield", "wisp_wand", "stoneguard_plate"].includes(item.itemId));
  const body = new THREE.Mesh(new THREE.SphereGeometry(0.45, 10, 8), mat(rare ? 0xd08cff : 0xf0c05a, 0.92, rare));
  const beam = new THREE.Mesh(new THREE.CylinderGeometry(0.05, 0.08, rare ? 4.6 : 2.6, 8), mat(rare ? 0xd08cff : 0xffdc88, 0.38, true));
  beam.position.y = rare ? 2.2 : 1.2;
  group.add(body, beam);
  group.position.set(bag.position.x, bag.position.y || 0.25, bag.position.z);
  group.id = bag.id;
  return group;
}

function findNearestEnemy(range) {
  const player = groupPosition(game.playerGroup);
  let nearest = null;
  let best = Infinity;
  for (const enemy of game.enemies.values()) {
    const dist = distance2d(player, enemy.position);
    if (dist < range && dist < best) {
      nearest = enemy;
      best = dist;
    }
  }
  return nearest;
}

function findNearestLoot(range) {
  const player = groupPosition(game.playerGroup);
  let nearest = null;
  let best = Infinity;
  for (const bag of game.loot.values()) {
    const dist = distance2d(player, bag.position);
    if (dist < range && dist < best) {
      nearest = bag;
      best = dist;
    }
  }
  return nearest;
}

function equipItem(itemId) {

function createAudioManager() {
  const context = new AudioContext();
  let musicOsc = null;
  const manager = {
    context,
    play(type) {
      if (game.save.settings.master <= 0) return;
      const osc = context.createOscillator();
      const gain = context.createGain();
      const now = context.currentTime;
      const profiles = {
        tick: [420, 0.05],
        low: [120, 0.12],
        swing: [260, 0.08],
        hit: [90, 0.12],
        hurt: [70, 0.18],
        pickup: [620, 0.12],
        level: [880, 0.3],
        pulse: [180, 0.25],
        dash: [320, 0.08],
        danger: [55, 0.24],
        equip: [540, 0.1],
        boss: [110, 0.5]
      };
      const [freq, duration] = profiles[type] || profiles.tick;
      osc.frequency.setValueAtTime(freq, now);
      osc.frequency.exponentialRampToValueAtTime(freq * 1.8, now + duration);
      gain.gain.setValueAtTime(0.0001, now);
      gain.gain.exponentialRampToValueAtTime(0.18 * game.save.settings.master * game.save.settings.sfx, now + 0.015);
      gain.gain.exponentialRampToValueAtTime(0.0001, now + duration);
      osc.connect(gain).connect(context.destination);
      osc.start(now);
      osc.stop(now + duration + 0.02);
    },
    ensureMusic() {
      if (musicOsc || game.save.settings.music <= 0) return;
      musicOsc = context.createOscillator();
      const gain = context.createGain();
      musicOsc.type = "sine";
      musicOsc.frequency.value = game.zone === ZONES.BOSS ? 74 : 110;
      gain.gain.value = 0.018 * game.save.settings.master * game.save.settings.music;
      musicOsc.connect(gain).connect(context.destination);
      musicOsc.start();
    }
  };
  return manager;
}

function playUiSound(type) {
  if (game.audio) {
    if (game.audio.context.state === "suspended") game.audio.context.resume();
    game.audio.play(type);
    game.audio.ensureMusic();
  }
}

function pulseEffect() {
  const ring = new THREE.Mesh(new THREE.RingGeometry(0.2, 0.35, 32), mat(0x77d8ff, 0.65, true));
  ring.rotation.x = -Math.PI / 2;
  ring.position.copy(game.playerGroup.position);
  ring.position.y = 0.08;
  game.scene.add(ring);
  const started = performance.now();
  const animate = () => {
    const t = (performance.now() - started) / 360;
    ring.scale.setScalar(1 + t * 14);
    ring.material.opacity = Math.max(0, 0.65 - t * 0.65);
    if (t < 1) requestAnimationFrame(animate);
    else game.scene.remove(ring);
  };
  animate();
}

function swingWeapon() {
  const sword = game.playerGroup.children.find((child) => child.geometry?.type === "BoxGeometry");
  if (!sword) return;
  const original = sword.rotation.z;
  sword.rotation.z = -1.2;
  setTimeout(() => {
    sword.rotation.z = original;
  }, 120);
}

function floatingText(text, position, color) {
  const label = makeLabel(text, color);
  label.position.set(position.x, (position.y || 0) + 2.4, position.z);
  game.scene.add(label);
  const started = performance.now();
  const animate = () => {
    const t = (performance.now() - started) / 900;
    label.position.y += 0.015;
    label.material.opacity = Math.max(0, 1 - t);
    if (t < 1) requestAnimationFrame(animate);
    else game.scene.remove(label);
  };
  animate();
}

function makeLabel(text, color = "#ffffff") {
  const canvas = document.createElement("canvas");
  canvas.width = 512;
  canvas.height = 128;
  const ctx = canvas.getContext("2d");
  ctx.fillStyle = "rgba(0,0,0,0.55)";
  ctx.roundRect(24, 28, 464, 66, 16);
  ctx.fill();
  ctx.font = "700 34px system-ui";
  ctx.textAlign = "center";
  ctx.fillStyle = color;
  ctx.fillText(text, 256, 74);
  const texture = new THREE.CanvasTexture(canvas);
  const sprite = new THREE.Sprite(new THREE.SpriteMaterial({ map: texture, transparent: true, depthTest: false }));
  sprite.scale.set(1.9, 0.48, 1);
  return sprite;
}

function addWorldLabel(text, x, y, z) {
  const label = makeLabel(text, "#ffe9b8");
  label.userData.fixedLabel = true;
  label.position.set(x, y, z);
  game.world.add(label);
}

function fitLabels(group) {
  if (!group || !game.camera) return;
  for (const child of group.children) {
    if (!child.isSprite || child.userData.fixedLabel) continue;
    const world = new THREE.Vector3();
    child.getWorldPosition(world);
    const dist = game.camera.position.distanceTo(world);
    child.visible = game.save.settings.nameTags && dist > 4.6;
    const width = THREE.MathUtils.clamp(dist * 0.085, 0.7, 1.65);
    child.scale.set(width, width * 0.25, 1);
  }
}

function mat(color, opacity = 1, emissive = false) {
  const material = new THREE.MeshStandardMaterial({
    color,
    roughness: 0.72,
    metalness: 0.04,
    transparent: opacity < 1,
    opacity
  });
  if (emissive) {
    material.emissive = new THREE.Color(color);
    material.emissiveIntensity = 0.55;
  }
  return material;
}

function movePointToward(position, target, amount) {
  const dx = target.x - position.x;
  const dz = target.z - position.z;
  const length = Math.max(0.001, Math.sqrt(dx * dx + dz * dz));
  position.x += (dx / length) * amount;
  position.z += (dz / length) * amount;
}

function clampToZone(position) {
  const bounds = getZone(game.zone).bounds;
  position.x = THREE.MathUtils.clamp(position.x, -bounds, bounds);
  position.z = THREE.MathUtils.clamp(position.z, -bounds, bounds);
  if (game.zone === ZONES.BOSS) {
    const radius = Math.sqrt(position.x * position.x + position.z * position.z);
    if (radius > 25.5) {
      position.x = (position.x / radius) * 25.5;
      position.z = (position.z / radius) * 25.5;
    }
  }
}

function collides(position) {
  return game.blockers.some((box) => Math.abs(position.x - box.x) < box.halfX && Math.abs(position.z - box.z) < box.halfZ);
}

function groupPosition(group) {

}

function randomRange(min, max) {

function clone(value) {
  return JSON.parse(JSON.stringify(value));
}

function escapeHtml(value) {
  return String(value).replace(/[&<>"']/g, (char) => ({
    "&": "&amp;",
    "<": "&lt;",
    ">": "&gt;",
    '"': "&quot;",
    "'": "&#039;"
  })[char]);
}
