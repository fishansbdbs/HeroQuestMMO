    playUiSound("tick");
    game.promptAction();
  }
}

function basicAttack() {
  performAttack({ kind: "auto", source: "manual" });
}

function resolveHotbarSlot(baseSlot) {
  if (!baseSlot) return { slot: 0, action: "empty", name: "Empty", key: "", icon: "--", cooldownMs: 0, locked: true, tooltip: "Empty hotbar slot." };
  const assigned = (game.save.hotbar || [])[baseSlot.slot - 1];
  if ([3, 7, 8].includes(baseSlot.slot)) {
    const assignedAbilityId = assigned === "pulse" ? "hero_pulse" : assigned;
    const ability = ABILITIES[assignedAbilityId];
    if (ability) {
      return {
        ...baseSlot,
        action: ability.id,
        name: ability.name,
        icon: abilityIcon(ability),
        cooldownMs: ability.cooldownMs || 0,
        locked: false,
        requiresAbility: ability.id,
        tooltip: ability.description
      };
    }
    return { ...baseSlot, action: `empty${baseSlot.slot}`, name: "Empty", icon: "--", cooldownMs: 0, locked: false, tooltip: "Open the Spellbook with K to assign an ability." };
  }
  return baseSlot;
}

function handleHotbarSlot(slotNumber) {
  const slot = resolveHotbarSlot(HOTBAR_DEFS.find((entry) => entry.slot === slotNumber));
  if (!slot || slot.locked || slot.action.startsWith("empty")) {
    toast("That hotbar slot is empty.");
    return;
  }
  handleHotbarAction(slot.action);
}

function handleHotbarAction(action) {
  const slot = HOTBAR_DEFS.find((entry) => entry.action === action) || (ABILITIES[action] ? { action, locked: false } : null);
  if (!slot || slot.locked) return;
  if (isLocallyDead()) {
    toast("You are defeated. Wait for respawn.");
    return;
  }
  game.hotbarPressed = action;
  setTimeout(() => {
    if (game.hotbarPressed === action) game.hotbarPressed = null;
  }, 140);
  if (action === "auto") toggleAutoAttack();
  if (action === "slash") performAttack({ kind: "slash", source: "hotbar" });
  if (action === "hero_pulse") useHeroPulse();
  if (ABILITIES[action]?.trainerId && action !== "hero_pulse") useAssignedAbility(action);
  if (action === "guard") useGuard();
  if (action === "potion") usePotion();
  if (action === "dash") dash();
}

function abilityIcon(ability) {
  return ability.name
    .split(/\s+/)
    .map((part) => part[0])
    .join("")
    .slice(0, 2)
    .toUpperCase();
}

function toggleAutoAttack() {
  if (isLocallyDead()) return;
  if (!getCurrentTarget()) {
    const nearest = findNearestEnemy(PLAYER_LIMITS.meleeRange + 1.2);
    if (nearest) selectTarget(nearest);
    else if (game.zone === ZONES.BOSS && game.boss && !game.boss.defeated) selectTarget("shadow_wyrm");
    else if (game.zone === ZONES.PALACE && game.iceMage && !game.iceMage.defeated) selectTarget(ICE_MAGE_BOSS.id);
  }
  if (!getCurrentTarget()) {
    toast("Select a target first.");
    logCombat("No target in range for auto attack.");
    return;
  }
  game.autoAttack = !game.autoAttack;
  game.lastAutoAttackAt = 0;
  logCombat(game.autoAttack ? "Auto attack started." : "Auto attack stopped.");
  playUiSound("tick");
}

function updateAutoAttack(now) {
  if (!game.autoAttack) return;
  if (isLocallyDead()) {
    game.autoAttack = false;
    return;
  }
  const target = getCurrentTarget();
  if (!target) {
    game.autoAttack = false;
    return;
  }
  const maxRange = target.id === "shadow_wyrm" || target.id === ICE_MAGE_BOSS.id ? 8 : PLAYER_LIMITS.meleeRange;
  if (distance2d(groupPosition(game.playerGroup), target.position) > maxRange + 0.25) {
    game.autoAttack = false;
    logCombat("Auto attack stopped: target out of range.");
    return;
  }
  if (now - game.lastAutoAttackAt >= PLAYER_LIMITS.attackCooldownMs) {
    game.lastAutoAttackAt = now;
    performAttack({ kind: "auto", source: "auto" });
  }
}

function performAttack({ kind = "auto", source = "manual" } = {}) {
  if (isLocallyDead()) {
    if (source !== "auto") toast("You cannot attack while defeated.");
    return false;
  }
  const now = performance.now();
  const cooldownKey = kind === "slash" ? "slash" : "attack";
  const cooldownMs = kind === "slash" ? ABILITIES.slash.cooldownMs : PLAYER_LIMITS.attackCooldownMs;
  if (game.cooldowns[cooldownKey] && now - game.cooldowns[cooldownKey] < cooldownMs) return false;
  let target = getCurrentTarget();
  if (!target) {
    target = findNearestEnemy(PLAYER_LIMITS.meleeRange);
    if (target) selectTarget(target);
  }
  if (!target && game.zone === ZONES.BOSS && game.boss && !game.boss.defeated) {
    target = { id: "shadow_wyrm", name: BOSS.name, position: { x: 0, y: 0, z: 0 }, health: game.boss.health, maxHealth: game.boss.maxHealth, level: BOSS.level };
    selectTarget("shadow_wyrm");
  }
  if (!target && game.zone === ZONES.PALACE && game.iceMage && !game.iceMage.defeated) {
    target = { id: ICE_MAGE_BOSS.id, name: ICE_MAGE_BOSS.name, position: { x: 0, y: 0, z: 0 }, health: game.iceMage.health, maxHealth: game.iceMage.maxHealth, level: ICE_MAGE_BOSS.level };
    selectTarget(ICE_MAGE_BOSS.id);
  }
  if (!target) {
    if (source !== "auto") logCombat("No target selected.");
    return false;
  }
  const maxRange = target.id === "shadow_wyrm" || target.id === ICE_MAGE_BOSS.id ? 8 : PLAYER_LIMITS.meleeRange;
  if (distance2d(groupPosition(game.playerGroup), target.position) > maxRange) {
    if (source !== "auto") logCombat(`${targetDisplayName(target)} is out of range.`);
    return false;
  }

  game.cooldowns[cooldownKey] = now;
  markCombat();
  if (kind === "slash") game.cooldowns.attack = now;
  playUiSound(kind === "slash" ? "slash" : "swing");
  swingWeapon(kind);
  slashTrail(kind);
  const scale = kind === "slash" ? ABILITIES.slash.damageScale : 1;
  const damage = Math.max(1, Math.round(calculatePlayerDamage(game.save) * scale));
  if (target.localOnly) {
    damageLocalEnemy(target.id, damage, { kind });
    return true;
  }
  if (game.online) {
    game.socket.emit(NET.COMBAT_ATTACK, { targetId: target.id, kind: kind === "slash" ? "slash" : "auto" }, handleCombatAck);
    logCombat(`${kind === "slash" ? "Slash" : "Auto Attack"} sent at ${targetDisplayName(target)}.`);
    return true;
  }
  if (target.id === "shadow_wyrm") hitBoss(damage, kind);
  else if (target.id === ICE_MAGE_BOSS.id) hitIceMage(damage, kind);
  else damageLocalEnemy(target.id, damage, { kind });
  return true;
}

function useHeroPulse() {
  if (isLocallyDead()) {
    toast("You cannot use abilities while defeated.");
    return;
  }
  if (!knowsAbility("hero_pulse")) {
    toast("Learn Hero Pulse from the Mage Trainer first.");
    logCombat("Hero Pulse is not learned.");
    return;
  }
  const now = performance.now();
  const weapon = getItem(game.save.equippedWeapon);
  const cooldown = ABILITIES.hero_pulse.cooldownMs * (weapon?.pulseCooldownBonus || 1);
  if (game.cooldowns.pulse && now - game.cooldowns.pulse < cooldown) return;
  const manaSpend = spendMana(game.save, ABILITIES.hero_pulse.manaCost || 0);
  if (!manaSpend.ok) {
    toast("Not enough mana.");
    logCombat("Not enough mana for Hero Pulse.");
    return;
  }
  game.save = manaSpend.player;
  game.cooldowns.pulse = now;
  markCombat();
  playUiSound("pulse");
  pulseEffect();
  logCombat("Hero Pulse released.");
  let hitLocalOnly = false;
  for (const enemy of game.enemies.values()) {
    if (enemy.localOnly && distance2d(enemy.position, groupPosition(game.playerGroup)) <= ABILITIES.hero_pulse.radius) {
      hitLocalOnly = true;
      damageLocalEnemy(enemy.id, Math.round(calculatePlayerDamage(game.save) * ABILITIES.hero_pulse.damageScale), { kind: "pulse" });
    }
  }
  if (game.online && !hitLocalOnly) {
    game.socket.emit(NET.COMBAT_ABILITY, {}, handleCombatAck);
    return;
  }
  for (const enemy of game.enemies.values()) {
    if (enemy.localOnly) continue;
    if (distance2d(enemy.position, groupPosition(game.playerGroup)) <= ABILITIES.hero_pulse.radius) {
      damageLocalEnemy(enemy.id, Math.round(calculatePlayerDamage(game.save) * ABILITIES.hero_pulse.damageScale), { kind: "pulse" });
    }
  }
  if (game.zone === ZONES.BOSS && game.boss && distance2d(groupPosition(game.playerGroup), { x: 0, z: 0 }) <= 8) {
    hitBoss(Math.round(calculatePlayerDamage(game.save) * ABILITIES.hero_pulse.damageScale), "pulse");
  }
  if (game.zone === ZONES.PALACE && game.iceMage && distance2d(groupPosition(game.playerGroup), { x: 0, z: 0 }) <= 8) {
    hitIceMage(Math.round(calculatePlayerDamage(game.save) * ABILITIES.hero_pulse.damageScale), "pulse");
  }
  saveGame();
}

function useAssignedAbility(abilityId) {
  const ability = ABILITIES[abilityId];
  if (!ability) return;
  if (isLocallyDead()) {
    toast("You cannot use abilities while defeated.");
    return;
  }
  if (!knowsAbility(abilityId)) {
    toast("That ability is not learned.");
    logCombat(`${ability.name} is not learned.`);
    return;
  }
  const now = performance.now();
  if (game.cooldowns[abilityId] && now - game.cooldowns[abilityId] < ability.cooldownMs) return;
  if (!game.online) {
    useAssignedAbilityLocal(abilityId, ability, now);
    return;
  }
  const target = getCurrentTarget();
  game.socket.emit(NET.COMBAT_ABILITY, { abilityId, targetId: target?.id }, (ack) => {
    if (!ack?.ok) {
      toast(abilityUseReason(ack?.reason, ability));
      logCombat(abilityUseReason(ack?.reason, ability));
      return;
    }
    game.cooldowns[abilityId] = performance.now();
    handleCombatAck(ack);
    for (const heal of ack.heals || []) {
      if (heal.playerId === game.socket?.id) {
        floatingText(`+${heal.amount}`, groupPosition(game.playerGroup), "#6df08a");
      }
    }
    saveGame();
  });
}

function useAssignedAbilityLocal(abilityId, ability, now = performance.now()) {
  const manaSpend = spendMana(game.save, ability.manaCost || 0);
  if (!manaSpend.ok) {
    toast("Not enough mana.");
    logCombat(`Not enough mana for ${ability.name}.`);
    return;
  }
  const target = getCurrentTarget();
  const playerPos = groupPosition(game.playerGroup);
  game.save = manaSpend.player;
  game.cooldowns[abilityId] = now;
  markCombat();

  if (["healing_orb", "mend_ally", "radiant_ward", "hearth_ward", "static_renewal"].includes(abilityId)) {
    useSupportAbilityLocal(abilityId, ability, playerPos);
    saveGame();
    return;
  }

  if (["ground_pound", "whirlwind_cleave", "magma_breaker", "flame_wave", "thunder_leap"].includes(abilityId)) {
    const radius = ability.radius || 5;
    areaEffect({ type: abilityId, origin: playerPos, radius });
    const damage = Math.max(1, Math.round(calculatePlayerDamage(game.save) * (ability.damageScale || 1)));
    let hits = 0;
    for (const enemy of [...game.enemies.values()]) {
      if (enemy.health > 0 && distance2d(enemy.position, playerPos) <= radius) {
        damageLocalEnemy(enemy.id, damage, { kind: abilityId });
        hits += 1;
      }
    }
    if (game.zone === ZONES.BOSS && game.boss && distance2d(playerPos, { x: 0, z: 0 }) <= radius) hitBoss(damage, abilityId);
    if (game.zone === ZONES.PALACE && game.iceMage && distance2d(playerPos, { x: 0, z: 0 }) <= radius) hitIceMage(damage, abilityId);
    logCombat(`${ability.name} hit ${hits} target${hits === 1 ? "" : "s"}.`);
    saveGame();
    return;
  }

  if (!target) {
    toast("Select a target first.");
    logCombat(`${ability.name} needs a target.`);
    game.cooldowns[abilityId] = 0;
    game.save = { ...game.save, mana: Math.min(game.save.maxMana, (game.save.mana || 0) + (ability.manaCost || 0)) };
    return;
  }
  if (distance2d(playerPos, target.position) > (ability.range || PLAYER_LIMITS.meleeRange)) {
    toast("Target is out of range.");
    logCombat(`${targetDisplayName(target)} is out of range.`);
    return;
  }

  const damage = Math.max(1, Math.round(calculatePlayerDamage(game.save) * (ability.damageScale || 1)));
  if (abilityId === "dark_punch") meleeEffect({ type: abilityId, from: playerPos, to: target.position });
  else if (abilityId === "chain_frost" || abilityId === "storm_bolt") projectileEffect({ type: abilityId, from: playerPos, chainTargets: [{ position: target.position }] });
  else projectileEffect({ type: abilityId, from: playerPos, to: target.position });
  if (target.id === "shadow_wyrm") hitBoss(damage, abilityId);
  else if (target.id === ICE_MAGE_BOSS.id) hitIceMage(damage, abilityId);
  else damageLocalEnemy(target.id, damage, { kind: abilityId });
  logCombat(`${ability.name} hit ${targetDisplayName(target)} for ${damage}.`);
  saveGame();
}

function useSupportAbilityLocal(abilityId, ability, playerPos) {
  if (abilityId === "healing_orb") {
    const orbId = `local_orb_${Date.now()}`;
    const amount = Math.round(45 + (game.save.healingPower || 0) * 1.4);
    const orb = {
      id: orbId,
      zone: game.zone,
      amount,
      expiresAt: Date.now() + (ability.durationMs || 10000),
      position: { x: playerPos.x + 1.4, y: 0.45, z: playerPos.z }
    };
    game.healingOrbs.set(orbId, orb);
    healingOrbEffect(orb);
    toast("Healing Orb created.");
    return;
  }
  const healAmount = ["mend_ally"].includes(abilityId)
    ? Math.round(50 + (game.save.healingPower || 0) * 1.8)
    : abilityId === "static_renewal"
      ? Math.round((ability.heal || 36) + (game.save.healingPower || 0) * 0.9)
      : 0;
  if (healAmount > 0) {
    const before = game.save.health || 0;
    game.save.health = Math.min(game.save.maxHealth || before, before + healAmount);
    floatingText(`+${Math.max(0, game.save.health - before)}`, playerPos, "#6df08a");
    healingBeamEffect({ type: abilityId, from: playerPos, to: playerPos, amount: healAmount });
    toast(`${ability.name} restored health.`);
    return;
  }
  game.guardUntil = Math.max(game.guardUntil || 0, performance.now() + (ability.durationMs || 3500));
  areaEffect({ type: abilityId, origin: playerPos, radius: ability.radius || 6 });
  shieldEffect();
  toast(`${ability.name} active.`);
}

function abilityUseReason(reason, ability) {
  if (reason === "not_learned") return `${ability.name} is not learned.`;
  if (reason === "mana") return "Not enough mana.";
  if (reason === "cooldown") return `${ability.name} is on cooldown.`;
  if (reason === "target") return "Wrong target type.";
  if (reason === "range") return "Target is out of range.";
  if (reason === "full") return "Target health is already full.";
  if (reason === "unsupported") return `${ability.name} is not combat-ready yet.`;
  return `${ability.name} failed.`;
}

function markCombat() {
  game.lastCombatAt = performance.now();
}

function updateMana(dt) {
  if (isLocallyDead()) return;
  game.save = regenerateMana(game.save, dt, performance.now() - (game.lastCombatAt || 0) < 5000);
}

function useGuard() {
  if (isLocallyDead()) return;
  const now = performance.now();
  if (game.cooldowns.guard && now - game.cooldowns.guard < ABILITIES.guard.cooldownMs) return;
  game.cooldowns.guard = now;
  game.guardUntil = now + ABILITIES.guard.durationMs;
  shieldEffect();
  playUiSound("guard");
  logCombat("Guard raised.");
}

function dash() {
  if (isLocallyDead()) return;
  const now = performance.now();
  if (game.cooldowns.dash && now - game.cooldowns.dash < ABILITIES.dash.cooldownMs) return;
  game.cooldowns.dash = now;
  const forward = new THREE.Vector3(-Math.sin(game.playerGroup.rotation.y), 0, -Math.cos(game.playerGroup.rotation.y));
  dashTrail();
  game.playerGroup.position.addScaledVector(forward, ABILITIES.dash.distance);
  clampToZone(game.playerGroup.position);
  playUiSound("dash");
  logCombat("Dash.");
}

function usePotion() {
  if (isLocallyDead()) {
    toast("You cannot drink potions while defeated.");
    return;
  }
  const now = performance.now();
  if (game.cooldowns.potion && now - game.cooldowns.potion < ABILITIES.potion.cooldownMs) return;
  if ((game.save.health ?? 0) >= (game.save.maxHealth ?? 1)) {
    toast("Health is already full.");
    logCombat("Health is already full.");
    return;
  }
  if (game.potionRequestPending) return;
  if (game.online) {
    game.potionRequestPending = true;
    game.socket.emit(NET.PLAYER_USE_CONSUMABLE, { itemId: "small_health_potion" }, (ack) => {
      game.potionRequestPending = false;
      if (!ack?.ok) {
        toast(potionUseReason(ack?.reason));
        logCombat(potionUseReason(ack?.reason));
        return;
      }
      game.cooldowns.potion = performance.now();
      if (ack.player) applyServerPlayerUpdate(ack.player);
      const amount = ack.heal?.amount || ABILITIES.potion.heal;
      floatingText(`+${amount}`, groupPosition(game.playerGroup), "#6df08a");
      playUiSound("pickup");
      toast("Health potion used.");
      logCombat(`Potion restored ${amount} HP.`);
      saveGame();
      if (!ui.panel?.classList.contains("hidden") && ui.panelTitle?.textContent === "Inventory") renderInventoryPanel();
    });
    return;
  }
  const result = consumeInventoryItem(game.save.inventory, "small_health_potion", 1);
  if (!result.consumed) {
    toast("No health potion available.");
    logCombat("No potion available.");
    return;
  }
  game.cooldowns.potion = now;
  game.save.inventory = result.inventory;
  const healAmount = ABILITIES.potion.heal + (game.save.potionBonus || 0);
  const before = game.save.health;
  game.save.health = Math.min(game.save.maxHealth, game.save.health + healAmount);
  const actual = Math.max(0, game.save.health - before);
  floatingText(`+${actual}`, groupPosition(game.playerGroup), "#6df08a");
  playUiSound("pickup");
  toast("Health potion used.");
  logCombat(`Potion restored ${actual} HP.`);
  saveGame();
}

function potionUseReason(reason) {
  if (reason === "full") return "Health is already full.";
  if (reason === "cooldown") return "Potion is on cooldown.";
  if (reason === "dead") return "You cannot drink potions while defeated.";
  if (reason === "missing") return "No health potion available.";
  return "Potion could not be used.";
}

function damageLocalEnemy(enemyId, damage, options = {}) {
  const enemy = game.enemies.get(enemyId);
  if (!enemy) return;
  enemy.health = Math.max(0, enemy.health - damage);
  const def = getEnemy(enemy.enemyId);
  floatingText(`-${damage}`, enemy.position, options.kind === "slash" ? "#ffe38b" : "#ffdd9a");
  hitSpark(enemy.position, options.kind);
  flashMesh(game.enemyMeshes.get(enemyId), 0xfff0a0);
  logCombat(`${options.kind === "slash" ? "Slash" : options.kind === "pulse" ? "Hero Pulse" : "Hit"} ${def.name} for ${damage}.`);
  playUiSound("hit");
  if (enemy.health <= 0) {
    if (enemy.enemyId === "training_dummy") {
      toast("Training dummy reset incoming.");
      logCombat("Training dummy reached 0 HP and will reset.");
      setTimeout(() => {
        const current = game.enemies.get(enemyId);
        if (current) current.health = current.maxHealth;
      }, 1800);
      return;
    }
    awardEnemy(def);
    spawnLocalLoot(def, enemy.position);
    startEnemyDeath(enemyId, def, enemy.position);
  }
}

function awardEnemy(enemyDef) {
  const loot = rollLoot(enemyDef);
  const beforeLevel = game.save.level;
  game.save = addProgressRewards(game.save, { xp: loot.xp, coins: loot.coins });
  toast(`+${loot.xp} XP, +${loot.coins} coins`);
  logCombat(`Reward: ${loot.xp} XP and ${loot.coins} coins.`);
  const questUpdate = applyQuestKill(game.save.questProgress, enemyDef);
  game.save.questProgress = questUpdate.progress;
  const bestiary = recordBestiaryKill(game.save, enemyDef, { elite: enemyDef.elite });
  if (bestiary.ok) game.save = bestiary.player;
  if (game.save.level > beforeLevel) {
    toast(`Level ${game.save.level}! Fully healed.`);
    levelUpEffect();
    playUiSound("level");
  }
  for (const quest of questUpdate.completed) {
    toast(`Quest complete: ${quest.name}`);
    game.save = addProgressRewards(game.save, quest.reward);
    if (quest.reward.itemId) {
      const inventoryResult = addInventoryStack(game.save.inventory, quest.reward.itemId, 1);
      if (inventoryResult.ok) game.save.inventory = inventoryResult.inventory;
      else toast("Inventory full. Quest item stayed pending.");
    }
    if (quest.reward.title) game.save.title = quest.reward.title;
  }
  refreshLocalMetaProgress();
  saveGame();
}

function spawnLocalLoot(enemyDef, position) {
  const drop = rollLoot(enemyDef);
  const id = `local_loot_${game.localLootId++}`;
  game.loot.set(id, {
    id,
    zone: game.zone,
    position: { ...position, y: 0.25 },
    coins: 0,
    items: drop.items
  });
}

function claimLoot(lootId) {
  if (isLocallyDead()) {
    toast("You cannot collect loot while defeated.");
    return;
  }
  if (game.online) {
    game.socket.emit(NET.LOOT_CLAIM, { lootId }, (ack) => {
      if (ack?.ok) {
        if (ack.player) applyServerPlayerUpdate(ack.player);
        else applyLootBag(ack.bag);
        game.loot.delete(lootId);
        toast("Loot collected.");
      } else {
        toast(lootClaimReason(ack?.reason));
      }
    });
    return;
  }
  const bag = game.loot.get(lootId);
  if (!bag) return toast("That loot is no longer available.");
  if (distance2d(groupPosition(game.playerGroup), bag.position) > 4) return toast("Move closer to collect that loot.");
  if (!applyLootBag(bag)) return;
