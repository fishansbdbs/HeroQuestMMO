function installDebugHooks() {
  if (typeof window === "undefined") return;
  const debugState = document.createElement("script");
  debugState.type = "application/json";
  debugState.id = "heroquest-debug-state";
  debugState.hidden = true;
  document.documentElement.appendChild(debugState);

  window.__heroQuestDebug = {
    snapshot() {
      const parts = game.playerGroup?.userData?.parts || {};
      const position = game.playerGroup ? groupPosition(game.playerGroup) : null;
      return {
        version: GAME_VERSION,
        started: game.started,
        zone: game.zone,
        position,
        rotationY: game.playerGroup?.rotation.y ?? 0,
        velocity: {
          x: game.playerVelocity.x,
          y: game.playerVelocity.y,
          z: game.playerVelocity.z
        },
        camera: game.camera
          ? {
              x: game.camera.position.x,
              y: game.camera.position.y,
              z: game.camera.position.z,
              yaw: game.input.yaw,
              pitch: game.input.pitch
            }
          : null,
        keys: [...game.input.keys],
        weapon: {
          hasRightHandAnchor: Boolean(parts.rightHandAnchor),
          hasWeaponGroup: Boolean(parts.weaponGroup),
          swordParentedToHand: Boolean(parts.sword && parts.sword.parent === parts.rightHandAnchor),
          swordRotation: parts.sword
            ? {
                x: parts.sword.rotation.x,
                y: parts.sword.rotation.y,
                z: parts.sword.rotation.z
              }
            : null
        },
        progression: {
          mana: game.save.mana,
          maxMana: game.save.maxMana,
          availableAttributePoints: game.save.availableAttributePoints,
          availableSkillPoints: game.save.availableSkillPoints,
          spentAttributes: game.save.spentAttributes
        },
        equipment: game.save.equipment,
        inventory: {
          slotsUsed: normalizeInventory(game.save.inventory || []).length,
          slotCount: INVENTORY_SLOT_COUNT
        },
        waypoints: game.save.waypoints || [],
        openedChests: game.save.openedChests || [],
        bestiaryProgress: game.save.bestiaryProgress || {},
        zoneCompletion: game.save.zoneCompletion || {},
        achievements: game.save.achievements || [],
        publicEvent: game.publicEvent,
        iceMage: game.iceMage,
        telegraphCount: game.telegraphs.length,
        enemyCount: game.enemies.size
      };
    },
    setKey(key, down = true) {
      if (down) game.input.keys.add(String(key).toLowerCase());
      else game.input.keys.delete(String(key).toLowerCase());
    },
    setCamera(yaw, pitch = game.input.pitch) {
      game.input.yaw = yaw;
      game.input.pitch = THREE.MathUtils.clamp(pitch, 0.22, 1.05);
      if (game.started) updateCamera();
    },
    action(action) {
      handleHotbarAction(action);
    }
  };

  const publishDebugState = () => {
    debugState.textContent = JSON.stringify(window.__heroQuestDebug.snapshot());
  };

  document.addEventListener("heroquest-debug-command", (event) => {
    const detail = event.detail || {};
    if (detail.type === "key") window.__heroQuestDebug.setKey(detail.key, detail.down);
    if (detail.type === "camera") window.__heroQuestDebug.setCamera(detail.yaw, detail.pitch);
    if (detail.type === "action") window.__heroQuestDebug.action(detail.action);
    publishDebugState();
  });

  setInterval(publishDebugState, 120);
  publishDebugState();
}

function bootstrap() {
  const params = new URLSearchParams(window.location.search);
  const smoke = params.get("smoke");
  game.debugDemo = params.get("demo");
  if (smoke) {
    const zone = smoke === "boss" ? ZONES.BOSS : smoke === "field" ? ZONES.FIELD : smoke === "frostveil" ? ZONES.FROSTVEIL : smoke === "palace" ? ZONES.PALACE : smoke === "vault" ? ZONES.FROSTBOUND_VAULT : ZONES.HUB;
    const phase2Smoke = params.get("phase2") === "progression";
    const phase4Smoke = params.get("phase4") === "trees";
    const phase5Smoke = params.get("phase5") === "equipment";
    const phase6Smoke = params.get("phase6") === "frostveil" || smoke === "frostveil";
    const phase7Smoke = params.get("phase7") === "ice_mage" || smoke === "palace";
    const phase8Smoke = params.get("phase8") === "meta";
    const smokeInventory = phase5Smoke
      ? [
          { itemId: "apprentice_staff", quantity: 1 },
          { itemId: "snowhide_helmet", quantity: 1 },
          { itemId: "frostguard_chestplate", quantity: 1 },
          { itemId: "small_health_potion", quantity: 5 }
        ]
      : game.save.inventory;
    game.save = applyEquipment({
      ...game.save,
      name: "SmokeTester",
      color: COLORS[1],
      xp: phase7Smoke ? 1300 : phase6Smoke ? 420 : phase2Smoke || phase5Smoke ? 260 : game.save.xp,
      coins: phase7Smoke ? 450 : phase2Smoke || phase5Smoke || phase6Smoke ? 150 : game.save.coins,
      inventory: phase2Smoke ? addInventoryItem(game.save.inventory || [], "rest_stone", 1) : smokeInventory,
      equipment: phase5Smoke ? createEquipmentState() : game.save.equipment,
      learnedAbilities: phase7Smoke || phase4Smoke ? ["hero_pulse"] : phase2Smoke ? [] : game.save.learnedAbilities,
      hotbar: phase7Smoke || phase4Smoke ? ["auto", "slash", "hero_pulse", "guard", "potion", "dash", null, null] : phase2Smoke ? ["auto", "slash", null, "guard", "potion", "dash", null, null] : game.save.hotbar,
      skillTreeNodes: phase4Smoke ? {} : game.save.skillTreeNodes,
      savedBuilds: phase4Smoke ? {} : game.save.savedBuilds,
      activeLoadout: phase4Smoke ? null : game.save.activeLoadout,
      bestiaryProgress: phase8Smoke ? {} : game.save.bestiaryProgress,
      zoneCompletion: phase8Smoke ? {} : game.save.zoneCompletion,
      achievements: phase8Smoke ? [] : game.save.achievements,
      waypoints: phase8Smoke ? ["dawnrest", "frostveil_camp"] : game.save.waypoints,
      openedChests: phase8Smoke ? ["frostveil_snow_cache"] : game.save.openedChests,
      zone,
      settings: { ...game.save.settings, showFps: true }
    });
    if (phase8Smoke) {
      game.save.learnedAbilities = ["hero_pulse"];
      game.save.firstClearRewards = { ...(game.save.firstClearRewards || {}), [ICE_MAGE_BOSS.id]: true };
      game.save.questProgress = applyQuestEvent(game.save.questProgress, "enter_frostveil").progress;
      game.save.questProgress = applyQuestEvent(game.save.questProgress, "activate_frost_ward").progress;
      game.save.questProgress = applyQuestEvent(game.save.questProgress, "activate_frost_ward").progress;
      game.save.questProgress = applyQuestEvent(game.save.questProgress, "activate_frost_ward").progress;
      game.save.questProgress = applyQuestEvent(game.save.questProgress, "enter_palace").progress;
      game.save.questProgress = applyQuestKill(game.save.questProgress, ICE_MAGE_BOSS).progress;
      for (const enemyId of ["frost_slime", "ice_goblin", "snow_wolf", "frost_wisp", "ice_golem", "frozen_knight"]) {

function loadSave() {
  const base = {
    name: "",
    color: COLORS[0],
    zone: ZONES.HUB,
    xp: STARTING_PLAYER.xp,
    coins: STARTING_PLAYER.coins,
    health: STARTING_PLAYER.health,
    inventory: clone(STARTER_INVENTORY),
    equipment: createEquipmentState({
      weapon: STARTING_PLAYER.equippedWeapon,
      chest: STARTING_PLAYER.equippedArmor
    }),
    equippedWeapon: STARTING_PLAYER.equippedWeapon,
    equippedArmor: STARTING_PLAYER.equippedArmor,
    openedChests: [],
    questProgress: createQuestProgress(),
    settings: {
      master: 0.8,
      music: 0.45,
      sfx: 0.75,
      sensitivity: 0.85,
      cameraDistance: 12,
      quality: "high",
      showFps: false,
      reducedMotion: false,
      screenShake: true,
      nameTags: true
    },
    title: ""
  };
  const finalizeSave = (incoming) => {
    const result = migrateFrostforgedSave(incoming);
    if (result.migrated) {
      try {
        localStorage.setItem(SAVE_BACKUP_KEY, JSON.stringify(result.backup));
      } catch {
        // Backup failure should not block a safe load.
      }
    }
    const loaded = recoverSavedHealth(applyEquipment({
      ...base,
      ...result.save,
      settings: { ...base.settings, ...(result.save.settings || {}) },
      questProgress: { ...base.questProgress, ...(result.save.questProgress || {}) },
      openedChests: result.save.openedChests || [],
      inventory: normalizeInventory(result.save.inventory || base.inventory),
      equipment: result.save.equipment || createEquipmentState({
        weapon: result.save.equippedWeapon || base.equippedWeapon,
        chest: result.save.equippedArmor || base.equippedArmor
      })
    }));
    loaded.migrationMessages = result.messages || [];
    return loaded;
  };
  try {
    const raw = localStorage.getItem(STORAGE_KEY);
    if (!raw) return finalizeSave(base);
    return finalizeSave(raw);
  } catch {
    return finalizeSave(base);
  }
}

function recoverSavedHealth(save) {
  const next = { ...save, state: save.state || "alive" };
  if ((next.health ?? 0) <= 0 || next.state === "dead") {
    next.state = "alive";
    next.zone = ZONES.HUB;
    next.health = next.maxHealth;
  }
  return next;
}

function refreshMenuServerStatus() {
  const status = uiRoot.querySelector("#menu-server-status");
  if (!status) return;
  fetch(`${SERVER_URL}/health`)
    .then((response) => {
      status.textContent = response.ok ? "Online" : "Solo ready";
      status.className = response.ok ? "online" : "offline";
    })
    .catch(() => {
      status.textContent = "Solo ready";
      status.className = "offline";
    });
}

function startGame() {
  uiRoot.innerHTML = "";
  buildHud();
  setupRenderer();
  setupInput();
  game.audio = createAudioManager();
  game.input.distance = game.save.settings.cameraDistance;
  game.zone = game.save.zone || ZONES.HUB;
  loadZone(game.zone, true);
  if (game.debugDemo === "frostward" && game.zone === ZONES.FROSTVEIL) {
    game.playerGroup.position.set(8, 0, -8);
  }
  setupNetwork();
  game.started = true;
  game.clock.start();
  requestAnimationFrame(loop);
  toast("Welcome to Dawnrest.");
  for (const message of game.save.migrationMessages || []) toast(message);
  delete game.save.migrationMessages;
  if (game.debugDemo === "combat") setTimeout(startCombatDemo, 650);
}

function setupInput() {
  if (setupInput.done) return;
  setupInput.done = true;
  window.addEventListener("keydown", (event) => {
    const key = event.key.toLowerCase();
    if (["w", "a", "s", "d", "shift", " "].includes(key)) event.preventDefault();
    game.input.keys.add(key);
    if (key === "i") openPanel("inventory");
    if (key === "c") openPanel("character");
    if (key === "k") openPanel("spellbook");
    if (key === "n") openPanel("skillTree");
    if (key === "b") openPanel("loadouts");
    if (key === "p") openPanel("party");
    if (key === "m") openPanel("quests");
    if (key === "y") openPanel("bestiary");
    if (key === "u") openPanel("achievements");
    if (key === "z") openPanel("zones");
    if (key === "h") openPanel("help");
    if (key === "escape") togglePanel("settings");
    if (key === "1") handleHotbarSlot(1);
    if (key === "2") handleHotbarSlot(2);
    if (key === "3") handleHotbarSlot(3);
    if (key === "4") handleHotbarSlot(4);
    if (key === "5") handleHotbarSlot(5);
    if (key === "6") handleHotbarSlot(6);
    if (key === "q") handleHotbarAction("pulse");
    if (key === "e") interact();
    if (key === " ") handleHotbarAction("dash");
  });
  window.addEventListener("keyup", (event) => game.input.keys.delete(event.key.toLowerCase()));
  window.addEventListener("pointerdown", (event) => {
    if (!game.started || event.target.closest("button,input,select,.panel")) return;
    game.input.dragging = event.button === 2 || event.buttons === 2;
    game.input.lastPointer = { x: event.clientX, y: event.clientY };
    if (event.button === 0) {
      const picked = pickEnemyFromPointer(event);
      if (picked) {
        selectTarget(picked);
        logCombat(`Targeted ${targetDisplayName(picked)}.`);
      } else {
        basicAttack();
      }
    }
  });
  window.addEventListener("pointerup", () => {
    game.input.dragging = false;
  });
  window.addEventListener("pointermove", (event) => {
    if (!game.input.dragging) return;
    const sensitivity = game.save.settings.sensitivity;
    game.input.yaw -= (event.clientX - game.input.lastPointer.x) * 0.006 * sensitivity;
    game.input.pitch = THREE.MathUtils.clamp(game.input.pitch + (event.clientY - game.input.lastPointer.y) * 0.003 * sensitivity, 0.22, 1.05);
    game.input.lastPointer = { x: event.clientX, y: event.clientY };
  });
  window.addEventListener("contextmenu", (event) => event.preventDefault());
}

function setupNetwork() {
  if (game.socket) game.socket.disconnect();
  game.socket = io(SERVER_URL, { timeout: 2500, transports: ["websocket", "polling"] });

  game.socket.on("connect", () => {
    game.online = true;
    updateNetworkStatus();
    game.socket.emit(NET.PLAYER_JOIN, serializeForServer(), (ack) => {
      if (ack?.snapshot) applySnapshot(ack.snapshot);
      toast("Online server connected.");
    });
  });

  game.socket.on("connect_error", () => {
    if (game.online) toast("Server unavailable. Continuing in solo mode.");
    game.online = false;
    updateNetworkStatus();
    if (game.enemies.size === 0 && game.zone === ZONES.FIELD) spawnLocalEnemies();
  });

  game.socket.on("disconnect", () => {
    game.online = false;
    updateNetworkStatus();
    toast("Disconnected. Solo local mode is active.");
  });

  game.socket.on(NET.WORLD_SNAPSHOT, applySnapshot);
  game.socket.on(NET.WORLD_EVENT, handleWorldEvent);
}

function loadZone(zoneId, moveToSpawn = false) {
  game.zone = zoneId;
  game.save.zone = zoneId;
  game.portals = [];
  game.npcs = [];
  game.chests.clear();
  game.blockers = [];
  game.enemies.clear();
  game.loot.clear();
  game.targetId = null;
  game.autoAttack = false;
  if (game.targetRing) {
    game.scene.remove(game.targetRing);
    game.targetRing = null;
  }
  clearMeshes(game.enemyMeshes);
  clearMeshes(game.lootMeshes);
  game.telegraphs.forEach((mesh) => game.scene.remove(mesh));
  game.telegraphs = [];
  if (game.world) game.scene.remove(game.world);
  game.boss = null;
  game.bossGroup = null;
  game.iceMage = null;
  game.iceMageGroup = null;
  game.world = new THREE.Group();
  game.scene.add(game.world);

  if (zoneId === ZONES.HUB) buildHub();
  if (zoneId === ZONES.FIELD) buildField();
  if (zoneId === ZONES.FROSTVEIL) buildFrostveil();
  if (zoneId === ZONES.FROSTBOUND_VAULT) buildFrostboundVault();
  if (zoneId === ZONES.PALACE) buildPalace();
  if (zoneId === ZONES.BOSS) buildBossArena();

  const zone = getZone(zoneId);
  if (moveToSpawn) {
    game.playerGroup.position.set(zone.spawn.x, 0, zone.spawn.z);
  }
  if (zoneId === ZONES.FROSTVEIL) applyZoneEntryProgress("enter_frostveil", "frostveil_camp");
  if (zoneId === ZONES.PALACE) applyZoneEntryProgress("enter_palace");
  if ((zoneId === ZONES.FIELD || zoneId === ZONES.FROSTVEIL) && !game.online) spawnLocalEnemies();
  if (zoneId === ZONES.BOSS) startLocalBoss();
  if (zoneId === ZONES.PALACE) startLocalIceMage();
  saveGame();
  updateHud();
  if (game.online) {
    game.socket.emit(NET.PLAYER_ZONE, { zone: zoneId, position: groupPosition(game.playerGroup) }, (ack) => {
      if (!ack?.ok) {
        toast(ack?.message || "That zone is locked.");
        return;
      }
      if (ack?.player) applyServerPlayerUpdate(ack.player);
      if (ack?.snapshot) applySnapshot(ack.snapshot);
    });
  }
}

function applyZoneEntryProgress(eventId, waypointId = null) {
  if (waypointId) {
    const waypoint = unlockWaypoint(game.save, waypointId);
    if (waypoint.ok) game.save = waypoint.player;
  }
  const questUpdate = applyQuestEvent(game.save.questProgress, eventId);
  game.save.questProgress = questUpdate.progress;
  for (const quest of questUpdate.completed) toast(`Quest complete: ${quest.name}`);
  refreshLocalMetaProgress();
}
