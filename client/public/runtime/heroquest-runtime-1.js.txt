function bootstrap() {
  const params = new URLSearchParams(window.location.search);
  const smoke = params.get("smoke");
  game.debugDemo = params.get("demo");
  if (smoke) {
    const zone = smoke === "boss" ? ZONES.BOSS : smoke === "field" ? ZONES.FIELD : ZONES.HUB;
    game.save = applyEquipment({
      ...game.save,
      name: "SmokeTester",
      color: COLORS[1],
      zone,
      settings: { ...game.save.settings, showFps: true }
    });
    startGame();
    return;
  }
  renderMainMenu();
}

function loadSave() {
  const base = {
    name: "",
    color: COLORS[0],
    zone: ZONES.HUB,
    xp: STARTING_PLAYER.xp,
    coins: STARTING_PLAYER.coins,
    health: STARTING_PLAYER.health,
    inventory: clone(STARTER_INVENTORY),
    equippedWeapon: STARTING_PLAYER.equippedWeapon,
    equippedArmor: STARTING_PLAYER.equippedArmor,
    questProgress: createQuestProgress(),
    settings: {
      master: 0.8,
      music: 0.45,
      sfx: 0.75,
      sensitivity: 0.85,
      cameraDistance: 12,
      quality: "high",
      showFps: false,
      reducedMotion: false,
      screenShake: true,
      nameTags: true
    },
    title: ""
  };
  try {
    const raw = localStorage.getItem(STORAGE_KEY);
    if (!raw) return applyEquipment(base);
    const parsed = JSON.parse(raw);
    return applyEquipment({
      ...base,
      ...parsed,
      settings: { ...base.settings, ...(parsed.settings || {}) },
      questProgress: { ...base.questProgress, ...(parsed.questProgress || {}) },
      inventory: parsed.inventory || base.inventory
    });
  } catch {
    return applyEquipment(base);
  }
}

function setupRenderer() {
  gameRoot.innerHTML = "";
  game.scene = new THREE.Scene();
  game.scene.background = new THREE.Color(0x121925);
  game.scene.fog = new THREE.Fog(0x141925, 38, 96);

  game.camera = new THREE.PerspectiveCamera(60, window.innerWidth / window.innerHeight, 0.1, 240);
  game.renderer = new THREE.WebGLRenderer({ antialias: true, powerPreference: "high-performance" });
  game.renderer.setSize(window.innerWidth, window.innerHeight);
  game.renderer.setPixelRatio(Math.min(window.devicePixelRatio, game.save.settings.quality === "low" ? 1.2 : 2));
  game.renderer.shadowMap.enabled = game.save.settings.quality !== "low";
  game.renderer.shadowMap.type = THREE.PCFSoftShadowMap;
  gameRoot.appendChild(game.renderer.domElement);

  const hemi = new THREE.HemisphereLight(0xfff4d6, 0x3b4760, 1.9);
  game.scene.add(hemi);
  const sun = new THREE.DirectionalLight(0xffdda0, 2.1);
  sun.position.set(22, 36, 18);
  sun.castShadow = true;
  sun.shadow.camera.left = -60;
  sun.shadow.camera.right = 60;
  sun.shadow.camera.top = 60;
  sun.shadow.camera.bottom = -60;
  game.scene.add(sun);

  game.playerGroup = createPlayerMesh(game.save.name, game.save.color, true);
  game.scene.add(game.playerGroup);

  window.addEventListener("resize", resize);
}

function setupInput() {
  if (setupInput.done) return;
  setupInput.done = true;
  window.addEventListener("keydown", (event) => {
    const key = event.key.toLowerCase();
    if (["w", "a", "s", "d", "shift", " "].includes(key)) event.preventDefault();
    game.input.keys.add(key);
    if (key === "i") openPanel("inventory");
    if (key === "c") openPanel("character");
    if (key === "p") openPanel("party");
    if (key === "m") openPanel("quests");
    if (key === "escape") togglePanel("settings");
    if (key === "1") handleHotbarAction("auto");
    if (key === "2") handleHotbarAction("slash");
    if (key === "3") handleHotbarAction("pulse");
    if (key === "4") handleHotbarAction("guard");
    if (key === "5") handleHotbarAction("potion");
    if (key === "6") handleHotbarAction("dash");
    if (key === "q") handleHotbarAction("pulse");
    if (key === "e") interact();
    if (key === " ") handleHotbarAction("dash");
  });
  window.addEventListener("keyup", (event) => game.input.keys.delete(event.key.toLowerCase()));
  window.addEventListener("pointerdown", (event) => {
    if (!game.started || event.target.closest("button,input,select,.panel")) return;
    game.input.dragging = event.button === 2 || event.buttons === 2;
    game.input.lastPointer = { x: event.clientX, y: event.clientY };
    if (event.button === 0) {
      const picked = pickEnemyFromPointer(event);
      if (picked) {
        selectTarget(picked);
        logCombat(`Targeted ${targetDisplayName(picked)}.`);
      } else {
        basicAttack();
      }
    }
  });
  window.addEventListener("pointerup", () => {
    game.input.dragging = false;
  });
  window.addEventListener("pointermove", (event) => {
    if (!game.input.dragging) return;
    const sensitivity = game.save.settings.sensitivity;
    game.input.yaw -= (event.clientX - game.input.lastPointer.x) * 0.006 * sensitivity;
    game.input.pitch = THREE.MathUtils.clamp(game.input.pitch + (event.clientY - game.input.lastPointer.y) * 0.003 * sensitivity, 0.22, 1.05);
    game.input.lastPointer = { x: event.clientX, y: event.clientY };
  });
  window.addEventListener("contextmenu", (event) => event.preventDefault());
}

function setupNetwork() {
  if (game.socket) game.socket.disconnect();
  game.socket = io(SERVER_URL, { timeout: 2500, transports: ["websocket", "polling"] });

  game.socket.on("connect", () => {
    game.online = true;
    updateNetworkStatus();
    game.socket.emit(NET.PLAYER_JOIN, serializeForServer(), (ack) => {
      if (ack?.snapshot) applySnapshot(ack.snapshot);
      toast("Online server connected.");
    });
  });

  game.socket.on("connect_error", () => {
    if (game.online) toast("Server unavailable. Continuing in solo mode.");
    game.online = false;
    updateNetworkStatus();
    if (game.enemies.size === 0 && game.zone === ZONES.FIELD) spawnLocalEnemies();
  });

  game.socket.on("disconnect", () => {
    game.online = false;
    updateNetworkStatus();
    toast("Disconnected. Solo local mode is active.");
  });

  game.socket.on(NET.WORLD_SNAPSHOT, applySnapshot);
}

function serializeForServer() {
  return {
    name: game.save.name,
    color: game.save.color,
    zone: game.zone,
    position: groupPosition(game.playerGroup),
    xp: game.save.xp,
    coins: game.save.coins,
    health: game.save.health,
    equippedWeapon: game.save.equippedWeapon,
    equippedArmor: game.save.equippedArmor,
    inventory: game.save.inventory,
    questProgress: game.save.questProgress,
    title: game.save.title
  };
}

function loadZone(zoneId, moveToSpawn = false) {
  game.zone = zoneId;
  game.save.zone = zoneId;
  game.portals = [];
  game.blockers = [];
  game.enemies.clear();
  game.loot.clear();
  game.targetId = null;
  game.autoAttack = false;
  if (game.targetRing) {
    game.scene.remove(game.targetRing);
    game.targetRing = null;
  }
  clearMeshes(game.enemyMeshes);
  clearMeshes(game.lootMeshes);
  game.telegraphs.forEach((mesh) => game.scene.remove(mesh));
  game.telegraphs = [];
  if (game.world) game.scene.remove(game.world);
  game.world = new THREE.Group();
  game.scene.add(game.world);

  if (zoneId === ZONES.HUB) buildHub();
  if (zoneId === ZONES.FIELD) buildField();
  if (zoneId === ZONES.BOSS) buildBossArena();

  const zone = getZone(zoneId);
  if (moveToSpawn) {
    game.playerGroup.position.set(zone.spawn.x, 0, zone.spawn.z);
  }
  if (zoneId === ZONES.FIELD && !game.online) spawnLocalEnemies();
  if (zoneId === ZONES.BOSS) startLocalBoss();
  saveGame();
  updateHud();
  if (game.online) {
    game.socket.emit(NET.PLAYER_ZONE, { zone: zoneId, position: groupPosition(game.playerGroup) }, (ack) => {
      if (ack?.snapshot) applySnapshot(ack.snapshot);
    });
  }
}
