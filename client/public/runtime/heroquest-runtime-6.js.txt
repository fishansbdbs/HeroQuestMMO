  if (snapshot.chests) {
    for (const chestState of snapshot.chests) {
      const chest = game.chests.get(chestState.id);
      if (chest) setChestOpened(chest, chestState.openedBy?.includes(game.socket?.id));
    }
  }
  syncRemotePlayers(snapshot.players || []);
  updatePartyList(snapshot.parties || [], snapshot.players || []);
}

function handleWorldEvent(event) {
  if (!event) return;
  if (event.type === "boss_telegraph" && game.zone === ZONES.BOSS) {
    game.boss = { ...game.boss, attack: event.attack };
    showTelegraph(event.attack);
    playUiSound("danger");
    logCombat(`${event.attack?.name || bossAttackText(event.attack?.type)} incoming.`);
  }
  if (event.type === "boss_impact" && game.zone === ZONES.BOSS) {
    bossImpactEffect(event.attack);
    const hit = event.hits?.find((entry) => entry.playerId === game.socket?.id);
    if (hit) {
      game.save.health = hit.health;
      floatingText(`-${hit.damage}`, groupPosition(game.playerGroup), "#ff8080");
      if (hit.health <= 0) beginDefeat();
    }
  }
  if (event.type === "boss_summon" && game.zone === ZONES.BOSS) {
    bossImpactEffect(event.attack);
    toast("The Shadow Wyrm summons shadow slimes.");
  }
  if (event.type === "boss_defeated") {
    if (!event.playerIds || event.playerIds.includes(game.socket?.id)) completeBossQuestLocal(event.reward || { xp: BOSS.xp, coins: BOSS.coins });
    clearTelegraphs();
    toast("Shadow Wyrm defeated!");
  }
  if (event.type === "player_defeated" && event.playerId === game.socket?.id) {
    beginDefeat(event.respawnAt || Date.now() + 5000);
  }
  if (event.type === "player_respawned" && event.playerId === game.socket?.id) {
    finishRespawn();
  }
}

function setChestOpened(chest, opened) {
  chest.opened = Boolean(opened);
  if (opened) {
    game.save.openedChests = Array.from(new Set([...(game.save.openedChests || []), chest.id]));
  }
  const lid = chest.mesh?.userData?.lid;
  if (lid) {
    lid.rotation.x = opened ? -0.95 : 0;
    lid.position.z = opened ? 0.28 : 0;
  }
}

function syncRemotePlayers(players) {
  const seen = new Set();
  for (const player of players) {
    if (player.id === game.socket?.id) {
      game.save.health = player.health ?? game.save.health;
      game.save.mana = player.mana ?? game.save.mana;
      game.save.xp = player.xp ?? game.save.xp;
      game.save.coins = player.coins ?? game.save.coins;
      game.save.level = player.level ?? game.save.level;
      game.save.maxHealth = player.maxHealth ?? game.save.maxHealth;
      game.save.maxMana = player.maxMana ?? game.save.maxMana;
      game.save.availableAttributePoints = player.availableAttributePoints ?? game.save.availableAttributePoints;
      game.save.availableSkillPoints = player.availableSkillPoints ?? game.save.availableSkillPoints;
      game.save.spentAttributes = player.spentAttributes || game.save.spentAttributes;
      game.save.inventory = player.inventory || game.save.inventory;
      game.save.learnedAbilities = player.learnedAbilities || game.save.learnedAbilities;
      game.save.hotbar = player.hotbar || game.save.hotbar;
      game.save.skillTreeNodes = player.skillTreeNodes || game.save.skillTreeNodes;
      game.save.state = player.state || game.save.state || "alive";
      if (player.respawnAt) {
        game.respawnAt = player.respawnAt;
        beginDefeat(player.respawnAt);
      } else if (game.save.state !== "dead") {
        game.respawnAt = null;
      }
      continue;
    }
    seen.add(player.id);
    let remote = game.remotePlayers.get(player.id);
    if (!remote) {
      const mesh = createPlayerMesh(player.name, player.color, false, player);
      game.scene.add(mesh);
      remote = { mesh, target: null };
      game.remotePlayers.set(player.id, remote);
    }
    applyVisualEquipment(remote.mesh, player);
    remote.target = new THREE.Vector3(player.position.x, 0, player.position.z);
    remote.mesh.visible = player.zone === game.zone && player.state !== "dead";
  }
  for (const [id, remote] of game.remotePlayers.entries()) {
    if (!seen.has(id)) {
      game.scene.remove(remote.mesh);
      game.remotePlayers.delete(id);
    }
  }
}

function syncEnemyMeshes() {

function createLootMesh(bag) {
  const group = new THREE.Group();
  const rare = bag.items?.some((item) => ["shadow_fang", "wyrm_scale_shield", "wisp_wand", "stoneguard_plate"].includes(item.itemId));
  const body = new THREE.Mesh(new THREE.SphereGeometry(0.45, 10, 8), mat(rare ? 0xd08cff : 0xf0c05a, 0.92, rare));
  const beam = new THREE.Mesh(new THREE.CylinderGeometry(0.05, 0.08, rare ? 4.6 : 2.6, 8), mat(rare ? 0xd08cff : 0xffdc88, 0.38, true));
  beam.position.y = rare ? 2.2 : 1.2;
  group.add(body, beam);
  group.position.set(bag.position.x, bag.position.y || 0.25, bag.position.z);
  group.userData.lootId = bag.id;
  return group;
}

function findNearestEnemy(range) {
  const player = groupPosition(game.playerGroup);
  let nearest = null;
  let best = Infinity;
  for (const enemy of game.enemies.values()) {
    const dist = distance2d(player, enemy.position);
    if (dist < range && dist < best) {
      nearest = enemy;
      best = dist;
    }
  }
  return nearest;
}

function equipItem(itemId) {
  const item = getItem(itemId);
  if (!item) return;
  if (item.type === "weapon") game.save.equippedWeapon = itemId;
  if (item.type === "armor") game.save.equippedArmor = itemId;
  game.save = applyEquipment(game.save);
  applyVisualEquipment(game.playerGroup, game.save);
  saveGame();
  playUiSound("equip");
  toast(`Equipped ${item.name}.`);
  renderInventoryPanel();
}

function ownsItem(itemId) {
  return game.save.equippedWeapon === itemId || game.save.equippedArmor === itemId || countItem(itemId) > 0 || itemId === "wooden_sword";
}

function buyShopItem(itemId) {
  if (isLocallyDead()) return toast("You cannot shop while defeated.");
  const stock = SHOP_ITEMS.find((entry) => entry.itemId === itemId);
  const item = getItem(itemId);
  if (!stock || !item) return;
  if (item.type !== "consumable" && ownsItem(itemId)) {
    toast(`${item.name} is already owned.`);
    return renderShopPanel();
  }
  if ((game.save.coins || 0) < stock.price) {
    toast(`Need ${stock.price} coins for ${item.name}.`);
    return;
  }
  game.save.coins -= stock.price;
  game.save.inventory = addInventoryItem(game.save.inventory, itemId, 1);
  game.save = applyEquipment(game.save);
  saveGame();
  playUiSound("pickup");
  toast(`Bought ${item.name}.`);
  renderShopPanel();
}

function openChest(chestId) {
  if (isLocallyDead()) return toast("You cannot open chests while defeated.");
  const chest = game.chests.get(chestId);
  if (!chest) return toast("That chest is gone.");
  if (chest.opened) return toast("That chest is already empty.");
  if (game.online) {
    game.socket.emit(NET.CHEST_CLAIM, { chestId }, (ack) => {
      if (!ack?.ok) return toast(chestClaimReason(ack?.reason));
      applyChestReward(chest, ack.reward);
    });
    return;
  }
  const reward = localChestReward(chestId);
  applyChestReward(chest, reward);
}

function localChestReward(chestId) {
  if (chestId === "field_south_cache") return { coins: 36, items: [{ itemId: "rusty_blade", quantity: 1 }] };
  if (chestId === "field_north_cache") return { coins: 28, items: [{ itemId: "small_health_potion", quantity: 1 }] };
  return { coins: 18, items: [{ itemId: "small_health_potion", quantity: 1 }] };
}

function applyChestReward(chest, reward) {
  setChestOpened(chest, true);
  game.save.coins += reward?.coins || 0;
  for (const item of reward?.items || []) {
    game.save.inventory = addInventoryItem(game.save.inventory, item.itemId, item.quantity || 1);
  }
  saveGame();
  playUiSound("pickup");
  toast(`Chest opened: +${reward?.coins || 0} coins${reward?.items?.length ? " and item loot" : ""}.`);
}

function chestClaimReason(reason) {

function applyServerPlayerUpdate(player) {
  if (!player) return;
  Object.assign(game.save, {
    health: player.health ?? game.save.health,
    maxHealth: player.maxHealth ?? game.save.maxHealth,
    mana: player.mana ?? game.save.mana,
    maxMana: player.maxMana ?? game.save.maxMana,
    level: player.level ?? game.save.level,
    xp: player.xp ?? game.save.xp,
    coins: player.coins ?? game.save.coins,
    spentAttributes: player.spentAttributes || game.save.spentAttributes,
    availableAttributePoints: player.availableAttributePoints ?? game.save.availableAttributePoints,
    availableSkillPoints: player.availableSkillPoints ?? game.save.availableSkillPoints,
    inventory: player.inventory || game.save.inventory,
    learnedAbilities: player.learnedAbilities || game.save.learnedAbilities,
    hotbar: player.hotbar || game.save.hotbar,
    skillTreeNodes: player.skillTreeNodes || game.save.skillTreeNodes
  });

}

function allocateAttributePoint(attributeId) {
  if (isLocallyDead()) return toast("You cannot allocate stats while defeated.");
  if (game.online) {
    game.socket.emit(NET.PLAYER_SPEND_ATTRIBUTE, { attributeId, count: 1 }, (ack) => {
      if (!ack?.ok) return toast(ack?.reason === "points" ? "No attribute points available." : "That stat cannot be increased.");
      applyServerPlayerUpdate(ack.player);
      saveGame();
      playUiSound("tick");
      renderCharacterPanel();
    });
    return;
  }
  const result = spendAttributePoint(game.save, attributeId, 1);
  if (!result.ok) return toast(result.reason === "points" ? "No attribute points available." : "That stat cannot be increased.");
  game.save = result.player;
  saveGame();
  playUiSound("tick");
  renderCharacterPanel();
}

function useRestStoneItem() {
  if (isLocallyDead()) return toast("You cannot reset stats while defeated.");
  if (game.online) {
    game.socket.emit(NET.PLAYER_USE_REST_STONE, {}, (ack) => {
      if (!ack?.ok) return toast("No Rest Stone available.");
      applyServerPlayerUpdate(ack.player);
      saveGame();
      playUiSound("equip");
      toast("Rest Stone restored your points.");
      renderCharacterPanel();
    });
    return;
  }
  const result = useRestStone(game.save);
  if (!result.ok) return toast("No Rest Stone available.");
  game.save = result.player;
  saveGame();
  playUiSound("equip");
  toast("Rest Stone restored your points.");
  renderCharacterPanel();
}
