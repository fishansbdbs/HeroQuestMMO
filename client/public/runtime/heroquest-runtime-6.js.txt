  if (snapshot.chests) {
    for (const chestState of snapshot.chests) {
      const chest = game.chests.get(chestState.id);
      if (chest) setChestOpened(chest, chestState.openedBy?.includes(game.socket?.id));
    }
  }
  syncRemotePlayers(snapshot.players || []);
  updatePartyList(snapshot.parties || [], snapshot.players || []);
}

function handleWorldEvent(event) {
  if (!event) return;
  if (event.type === "boss_telegraph" && game.zone === ZONES.BOSS) {
    game.boss = { ...game.boss, attack: event.attack };
    showTelegraph(event.attack);
    playUiSound("danger");
    logCombat(`${event.attack?.name || bossAttackText(event.attack?.type)} incoming.`);
  }
  if (event.type === "boss_impact" && game.zone === ZONES.BOSS) {
    bossImpactEffect(event.attack);
    const hit = event.hits?.find((entry) => entry.playerId === game.socket?.id);
    if (hit) {
      game.save.health = hit.health;
      floatingText(`-${hit.damage}`, groupPosition(game.playerGroup), "#ff8080");
      if (hit.health <= 0) beginDefeat();
    }
  }
  if (event.type === "boss_summon" && game.zone === ZONES.BOSS) {
    bossImpactEffect(event.attack);
    toast("The Shadow Wyrm summons shadow slimes.");
  }
  if (event.type === "boss_defeated") {
    if (!event.playerIds || event.playerIds.includes(game.socket?.id)) completeBossQuestLocal(event.reward || { xp: BOSS.xp, coins: BOSS.coins });
    clearTelegraphs();
    toast("Shadow Wyrm defeated!");
  }
  if (event.type === "ice_mage_telegraph" && game.zone === ZONES.PALACE) {
    game.iceMage = { ...game.iceMage, attack: event.attack };
    showTelegraph(event.attack);
    playUiSound("danger");
    logCombat(`${event.attack?.name || bossAttackText(event.attack?.type)} incoming.`);
  }
  if (event.type === "ice_mage_impact" && game.zone === ZONES.PALACE) {
    bossImpactEffect(event.attack);
    const hit = event.hits?.find((entry) => entry.playerId === game.socket?.id);
    if (hit) {
      game.save.health = hit.health;
      floatingText(`-${hit.damage}`, groupPosition(game.playerGroup), "#9eefff");
      if (hit.health <= 0) beginDefeat();
    }
  }
  if (event.type === "ice_mage_summon" && game.zone === ZONES.PALACE) {
    bossImpactEffect(event.attack);
    toast("Frozen servants answer Zero's call.");
  }
  if (event.type === "ice_mage_reset" && game.zone === ZONES.PALACE) {
    game.iceMage = { ...game.iceMage, active: false, attack: null };
    clearTelegraphs();
    toast("The Palace of Zero grows quiet.");
  }
  if (event.type === "ice_mage_defeated") {
    if (!event.playerIds || event.playerIds.includes(game.socket?.id)) completeIceMageQuestLocal(event.reward || { xp: ICE_MAGE_BOSS.xp, coins: ICE_MAGE_BOSS.coins });
    clearTelegraphs();
    toast("Zero, the Ice Mage defeated!");
  }
  if (event.type === "player_defeated" && event.playerId === game.socket?.id) {
    beginDefeat(event.respawnAt || Date.now() + 5000);
  }
  if (event.type === "player_respawned" && event.playerId === game.socket?.id) {
    finishRespawn();
  }
  if (event.type === "public_event_started" && event.eventId === "defend_frost_ward") {
    game.publicEvent = { ...game.publicEvent, ...event, active: true };
    toast("Public event: Defend the Frost Ward.");
  }
  if (event.type === "public_event_wave" && event.eventId === "defend_frost_ward") {
    toast(`Frost Ward wave ${event.wave} begins.`);
  }
  if (event.type === "public_event_completed" && event.eventId === "defend_frost_ward") {
    game.publicEvent = { ...game.publicEvent, active: false };
    toast("Frost Ward secured.");
  }
}

function setChestOpened(chest, opened) {
  chest.opened = Boolean(opened);
  if (opened) {
    game.save.openedChests = Array.from(new Set([...(game.save.openedChests || []), chest.id]));
  }
  const lid = chest.mesh?.userData?.lid;
  if (lid) {
    lid.rotation.x = opened ? -0.95 : 0;
    lid.position.z = opened ? 0.28 : 0;
  }
}

function syncRemotePlayers(players) {
  const seen = new Set();
  for (const player of players) {
    if (player.id === game.socket?.id) {
      game.save.health = player.health ?? game.save.health;
      game.save.mana = player.mana ?? game.save.mana;
      game.save.xp = player.xp ?? game.save.xp;
      game.save.coins = player.coins ?? game.save.coins;
      game.save.level = player.level ?? game.save.level;
      game.save.maxHealth = player.maxHealth ?? game.save.maxHealth;
      game.save.maxMana = player.maxMana ?? game.save.maxMana;
      game.save.availableAttributePoints = player.availableAttributePoints ?? game.save.availableAttributePoints;
      game.save.availableSkillPoints = player.availableSkillPoints ?? game.save.availableSkillPoints;
      game.save.spentAttributes = player.spentAttributes || game.save.spentAttributes;
      game.save.inventory = player.inventory || game.save.inventory;
      game.save.equipment = player.equipment || game.save.equipment;
      game.save.equippedWeapon = player.equippedWeapon || game.save.equippedWeapon;
      game.save.equippedArmor = player.equippedArmor || game.save.equippedArmor;
      game.save.magicPower = player.magicPower ?? game.save.magicPower;
      game.save.learnedAbilities = player.learnedAbilities || game.save.learnedAbilities;
      game.save.hotbar = player.hotbar || game.save.hotbar;
      game.save.skillTreeNodes = player.skillTreeNodes || game.save.skillTreeNodes;
      game.save.savedBuilds = player.savedBuilds || game.save.savedBuilds;
      game.save.activeLoadout = player.activeLoadout || game.save.activeLoadout;
      game.save.waypoints = player.waypoints || game.save.waypoints;
      game.save.openedChests = player.openedChests || game.save.openedChests;
      game.save.bestiaryProgress = player.bestiaryProgress || game.save.bestiaryProgress;
      game.save.zoneCompletion = player.zoneCompletion || game.save.zoneCompletion;
      game.save.achievements = player.achievements || game.save.achievements;
      game.save.firstClearRewards = player.firstClearRewards || game.save.firstClearRewards;
      game.save.title = player.title || game.save.title;
      game.save.state = player.state || game.save.state || "alive";
      if (player.respawnAt) {
        game.respawnAt = player.respawnAt;
        beginDefeat(player.respawnAt);
      } else if (game.save.state !== "dead") {
        game.respawnAt = null;
      }
      applyVisualEquipment(game.playerGroup, game.save);
      continue;
    }
    seen.add(player.id);
    let remote = game.remotePlayers.get(player.id);
    if (!remote) {
      const mesh = createPlayerMesh(player.name, player.color, false, player);
      game.scene.add(mesh);
      remote = { mesh, target: null };
      game.remotePlayers.set(player.id, remote);
    }
    applyVisualEquipment(remote.mesh, player);
    remote.target = new THREE.Vector3(player.position.x, 0, player.position.z);
    remote.mesh.visible = player.zone === game.zone && player.state !== "dead";
  }
  for (const [id, remote] of game.remotePlayers.entries()) {
    if (!seen.has(id)) {
      game.scene.remove(remote.mesh);
      game.remotePlayers.delete(id);
    }
  }
}

function createLootMesh(bag) {
  const group = new THREE.Group();
  const rare = bag.items?.some((item) => ["shadow_fang", "wyrm_scale_shield", "wisp_wand", "stoneguard_plate"].includes(item.itemId));
  const body = new THREE.Mesh(new THREE.SphereGeometry(0.45, 10, 8), mat(rare ? 0xd08cff : 0xf0c05a, 0.92, rare));
  const beam = new THREE.Mesh(new THREE.CylinderGeometry(0.05, 0.08, rare ? 4.6 : 2.6, 8), mat(rare ? 0xd08cff : 0xffdc88, 0.38, true));
  beam.position.y = rare ? 2.2 : 1.2;
  group.add(body, beam);
  group.position.set(bag.position.x, bag.position.y || 0.25, bag.position.z);
  group.userData.lootId = bag.id;
  return group;
}

function findNearestEnemy(range) {
  const player = groupPosition(game.playerGroup);
  let nearest = null;
  let best = Infinity;
  for (const enemy of game.enemies.values()) {
    const dist = distance2d(player, enemy.position);
    if (dist < range && dist < best) {
      nearest = enemy;
      best = dist;
    }
  }
  return nearest;
}

function equipItem(itemId) {
  const item = getItem(itemId);
  if (!item) return;
  const slot = slotForItem(item);
  if (!slot) return toast("That item cannot be equipped.");
  if (game.online) {
    game.socket.emit(NET.PLAYER_EQUIP_ITEM, { slot, itemId }, (ack) => {
      if (!ack?.ok) return toast(equipReason(ack?.reason));
      applyServerPlayerUpdate(ack.player);
      saveGame();
      playUiSound("equip");
      toast(`Equipped ${item.name}.`);
      renderInventoryPanel();
    });
    return;
  }
  const result = equipItemToSlot(game.save, slot, itemId);
  if (!result.ok) return toast(equipReason(result.reason));
  game.save = result.player;
  applyVisualEquipment(game.playerGroup, game.save);
  saveGame();
  playUiSound("equip");
  toast(`Equipped ${item.name}.`);
  renderInventoryPanel();
}

function equipmentSlotLabel(slot) {
  return slot.charAt(0).toUpperCase() + slot.slice(1);
}

function itemMatchesInventoryFilter(item, query, filter) {
  if (!item) return false;
  const searchable = `${item.name || ""} ${item.id || ""} ${item.description || ""}`.toLowerCase();
  if (query && !searchable.includes(query)) return false;
  if (filter === "equipment") return Boolean(slotForItem(item));
  if (filter === "consumable") return item.type === "consumable" || item.id === "rest_stone";
  if (filter === "material") return item.type === "material";
  return true;
}

function equipReason(reason) {
  if (reason === "level") return "Your level is too low for that item.";
  if (reason === "missing") return "That item is not in your inventory.";
  if (reason === "slot") return "That item does not fit that slot.";
  if (reason === "dead") return "You cannot equip items while defeated.";
  return "That item cannot be equipped.";
}

function renderShopPanel() {
  ui.panelTitle.textContent = "Weapon Shop";
  ui.panelContent.innerHTML = SHOP_ITEMS.map((stock) => {
    const item = getItem(stock.itemId);
    const owned = ownsItem(stock.itemId);
    const equipped = equippedItemIds().has(stock.itemId);
    const canEquip = Boolean(slotForItem(item));
    const stat = item?.attack ? `+${item.attack} attack` : item?.heal ? `Heals ${item.heal} HP` : "Starter item";
    const status = equipped ? "Equipped" : owned && item?.type !== "consumable" ? "Owned" : `${stock.price} coins`;
    return `<div class="item-row shop-row">
      <div><strong>${item?.name || stock.itemId}</strong><small>${item?.rarity || ""} ${stat}. ${item?.description || ""}</small></div>
      <div class="shop-actions">
        <span>${status}</span>
        ${canEquip ? `<button data-shop-equip="${stock.itemId}" ${owned ? "" : "disabled"}>Equip</button>` : ""}
        <button data-shop-buy="${stock.itemId}" ${equipped && item?.type !== "consumable" ? "disabled" : ""}>${owned && item?.type !== "consumable" ? "Owned" : "Buy"}</button>

  });

      equipItem(button.dataset.shopEquip);
      renderShopPanel();
    });

  ui.panelContent.querySelector("#close-shop").addEventListener("click", closePanel);

function ownsItem(itemId) {
  return equippedItemIds().has(itemId) || countItem(itemId) > 0 || itemId === "wooden_sword";
}

function equippedItemIds() {
  return new Set(Object.values(createEquipmentState({
    ...(game.save.equipment || {}),
    weapon: game.save.equipment?.weapon || game.save.equippedWeapon,
    chest: game.save.equipment?.chest || game.save.equippedArmor
  })).filter(Boolean));
}

function buyShopItem(itemId) {
  if (isLocallyDead()) return toast("You cannot shop while defeated.");
  const stock = SHOP_ITEMS.find((entry) => entry.itemId === itemId);
  const item = getItem(itemId);
  if (!stock || !item) return;
  if (item.type !== "consumable" && ownsItem(itemId)) {
    toast(`${item.name} is already owned.`);
    return renderShopPanel();
  }
  if ((game.save.coins || 0) < stock.price) {
    toast(`Need ${stock.price} coins for ${item.name}.`);
    return;
  }
  const inventoryResult = addInventoryStack(game.save.inventory, itemId, 1);
  if (!inventoryResult.ok) {
    toast("Inventory full. Make space before buying.");
    return renderShopPanel();
  }
  game.save.coins -= stock.price;
  game.save.inventory = inventoryResult.inventory;
  game.save = applyEquipment(game.save);
  saveGame();
  playUiSound("pickup");
  toast(`Bought ${item.name}.`);
  renderShopPanel();
}

function openChest(chestId) {
  if (isLocallyDead()) return toast("You cannot open chests while defeated.");
  const chest = game.chests.get(chestId);
  if (!chest) return toast("That chest is gone.");
  if (chest.opened) return toast("That chest is already empty.");
  if (game.online) {
    game.socket.emit(NET.CHEST_CLAIM, { chestId }, (ack) => {
      if (!ack?.ok) return toast(chestClaimReason(ack?.reason));
      applyChestReward(chest, ack.reward, ack.player);
    });
    return;
  }
  const reward = localChestReward(chestId);
  applyChestReward(chest, reward);
}

function localChestReward(chestId) {
  if (chestId === "frostveil_snow_cache") return { coins: 45, items: [{ itemId: "ice_shard", quantity: 2 }] };
  if (chestId === "field_south_cache") return { coins: 36, items: [{ itemId: "rusty_blade", quantity: 1 }] };
  if (chestId === "field_north_cache") return { coins: 28, items: [{ itemId: "small_health_potion", quantity: 1 }] };
  return { coins: 18, items: [{ itemId: "small_health_potion", quantity: 1 }] };
}

function applyChestReward(chest, reward, serverPlayer = null) {
  if (serverPlayer) {
    applyServerPlayerUpdate(serverPlayer);
    setChestOpened(chest, true);
    refreshLocalMetaProgress();
    saveGame();
    playUiSound("pickup");
    toast(`Chest opened: +${reward?.coins || 0} coins${reward?.items?.length ? " and item loot" : ""}.`);
    return true;
  }
  const inventoryResult = applyRewardItemsToInventory(game.save.inventory, reward?.items || []);
  if (!inventoryResult.ok) {
    toast(inventoryResult.reason === "full" ? "Inventory full. Make space before opening this chest." : "That chest cannot be opened.");
    return false;
  }
  setChestOpened(chest, true);
  game.save.inventory = inventoryResult.inventory;
  game.save.coins += reward?.coins || 0;
  refreshLocalMetaProgress();
  saveGame();
  playUiSound("pickup");
  toast(`Chest opened: +${reward?.coins || 0} coins${reward?.items?.length ? " and item loot" : ""}.`);
  return true;
}

function renderTrainerPanel(trainerId) {
  const title = trainerId === "healer" ? "Healer" : "Mage Trainer";
  ui.panelTitle.textContent = title;
  const abilities = getTrainerAbilities(trainerId);
  ui.panelContent.innerHTML = abilities.map((ability) => {
    const learned = knowsAbility(ability.id);
    const reqs = abilityRequirementText(ability);
    const status = learned ? "Learned" : `${ability.price} coins`;
    return `<div class="item-row">
      <div>
        <strong>${ability.name}</strong>
        <small>${ability.description} ${reqs}</small>
        <small>Scaling: ${ability.scalingStat || "none"} | Mana: ${ability.manaCost || 0} | Cooldown: ${Math.round((ability.cooldownMs || 0) / 1000)}s</small>
      </div>
      <div class="shop-actions">
        <span>${status}</span>
        <button data-train-ability="${ability.id}" ${learned ? "disabled" : ""}>Buy</button>
      </div>
    </div>`;
  }).join("") + `<button id="open-spellbook" class="primary">Open Spellbook</button>`;
  ui.panelContent.querySelectorAll("[data-train-ability]").forEach((button) => {
    button.addEventListener("click", () => buyTrainerAbility(button.dataset.trainAbility, trainerId));
  });
  ui.panelContent.querySelector("#open-spellbook").addEventListener("click", () => openPanel("spellbook"));
}

function abilityRequirementText(ability) {
  const attributes = Object.entries(ability.attributeRequirements || {})
    .map(([id, value]) => `${id} ${value}`)
    .join(", ");
  return `Requires level ${ability.levelRequirement || 1}${attributes ? `, ${attributes}` : ""}.`;
}

function buyTrainerAbility(abilityId, trainerId) {
  if (isLocallyDead()) return toast("You cannot train while defeated.");
  if (game.online) {
    game.socket.emit(NET.PLAYER_BUY_ABILITY, { abilityId }, (ack) => {
      if (!ack?.ok) return toast(trainerBuyReason(ack?.reason));
      applyServerPlayerUpdate(ack.player);
      saveGame();
      playUiSound("pickup");
      toast(`Learned ${ack.ability?.name || abilityId}.`);
      renderTrainerPanel(trainerId);
    });
    return;
  }
  const result = purchaseTrainerAbility(game.save, abilityId);
  if (!result.ok) return toast(trainerBuyReason(result.reason));
  game.save = result.player;
  saveGame();
  playUiSound("pickup");
  toast(`Learned ${result.ability.name}.`);
  renderTrainerPanel(trainerId);
}

function trainerBuyReason(reason) {
  if (reason === "known") return "That ability is already learned.";
  if (reason === "level") return "Your level is too low for that training.";
  if (reason === "attribute") return "Your attributes do not meet the requirement.";
  if (reason === "coins") return "Not enough coins.";
  return "That training is unavailable.";
}

function renderSpellbookPanel() {
  ui.panelTitle.textContent = "Spellbook";
  const abilities = ["hero_pulse", "ground_pound", "fireball", "dark_punch", "water_blast", "healing_orb", "mend_ally"]
    .map((abilityId) => ABILITIES[abilityId])
    .filter(Boolean);
  ui.panelContent.innerHTML = abilities.map((ability) => {
    const learned = knowsAbility(ability.id);
    const slotted = (game.save.hotbar || []).includes(ability.id);
    return `<div class="item-row">
      <div>
        <strong>${ability.name}</strong>
        <small>${learned ? "Learned" : "Locked"} from ${ability.trainerId === "healer" ? "Sister Elara" : "Arcanist Mira"}. ${ability.description}</small>
        <small>Mana ${ability.manaCost || 0}. ${abilityRequirementText(ability)}</small>
      </div>
      <div class="shop-actions">
        <span>${slotted ? "Slotted" : learned ? "Ready" : "Locked"}</span>
        ${ability.id === "hero_pulse" ? `<button data-slot-ability="${ability.id}" data-slot="3" ${learned ? "" : "disabled"}>${slotted ? "Slot 3" : "Assign 3"}</button>` : ""}
      </div>
    </div>`;
  }).join("");
  ui.panelContent.querySelectorAll("[data-slot-ability]").forEach((button) => {
    button.addEventListener("click", () => assignAbilityToHotbar(Number(button.dataset.slot), button.dataset.slotAbility));
  });
}

function assignAbilityToHotbar(slot, abilityId) {
  if (game.online) {
    game.socket.emit(NET.PLAYER_ASSIGN_HOTBAR, { slot, abilityId }, (ack) => {
      if (!ack?.ok) return toast("That ability cannot be assigned.");
      applyServerPlayerUpdate(ack.player);
      saveGame();
      toast("Hotbar updated.");
      renderSpellbookPanel();
    });
    return;
  }
  const result = assignHotbarAbility(game.save, slot, abilityId);
  if (!result.ok) return toast("That ability cannot be assigned.");
  game.save = result.player;
  saveGame();
  toast("Hotbar updated.");
  renderSpellbookPanel();
}

function renderSkillTreePanel() {
  const stats = applyEquipment(game.save);
  ui.panelTitle.textContent = "Skill Tree";
  ui.panelContent.innerHTML = `
    <div class="stat-line"><span>Skill Points</span><strong>${stats.availableSkillPoints}</strong></div>
    ${Object.values(SKILL_TREES).map((tree) => `
      <div class="quest-row"><strong>${tree.name}</strong><small>${tree.nodeIds.length} nodes</small></div>
      ${tree.nodeIds.map((nodeId) => {
        const node = SKILL_NODES[nodeId];
        const rank = stats.skillTreeNodes?.[nodeId] || 0;
        const prereq = (node.prerequisites || []).map((entry) => `${SKILL_NODES[entry.nodeId]?.name || entry.nodeId} ${entry.rank}`).join(", ");
        return `<div class="item-row">
          <div><strong>${node.name}</strong><small>${rank}/${node.maxRank}. ${node.description}${prereq ? ` Requires ${prereq}.` : ""}</small></div>
          <button class="mini-button" data-skill-node="${node.id}" ${rank >= node.maxRank || stats.availableSkillPoints <= 0 ? "disabled" : ""}>+</button>
        </div>`;
      }).join("")}
    `).join("")}
  `;
  ui.panelContent.querySelectorAll("[data-skill-node]").forEach((button) => {
    button.addEventListener("click", () => buySkillTreeNode(button.dataset.skillNode));
  });
}

function buySkillTreeNode(nodeId) {
  if (isLocallyDead()) return toast("You cannot train skill trees while defeated.");
  if (game.online) {
    game.socket.emit(NET.PLAYER_BUY_SKILL_NODE, { nodeId }, (ack) => {
      if (!ack?.ok) return toast(skillNodeReason(ack?.reason));
      applyServerPlayerUpdate(ack.player);
      saveGame();
      playUiSound("tick");
      renderSkillTreePanel();
    });
    return;
  }
  const result = purchaseSkillNode(game.save, nodeId);
  if (!result.ok) return toast(skillNodeReason(result.reason));
  game.save = result.player;
  saveGame();
  playUiSound("tick");
  renderSkillTreePanel();
}

function skillNodeReason(reason) {
  if (reason === "prerequisite") return "That node needs another node first.";
  if (reason === "points") return "No skill points available.";
  if (reason === "max_rank") return "That node is already maxed.";
  return "That skill node is unavailable.";
}

function renderLoadoutPanel() {
  ui.panelTitle.textContent = "Loadouts";
  const builds = game.save.savedBuilds || {};
  ui.panelContent.innerHTML = ["A", "B"].map((slot) => {
    const build = builds[slot];
    return `<div class="item-row">
      <div><strong>Loadout ${slot}</strong><small>${build?.label || "Empty build slot"}</small></div>
      <div class="shop-actions">
        <button data-save-loadout="${slot}">Save</button>
        <button data-activate-loadout="${slot}" ${build ? "" : "disabled"}>Activate</button>
      </div>
    </div>`;
  }).join("");
  ui.panelContent.querySelectorAll("[data-save-loadout]").forEach((button) => {
    button.addEventListener("click", () => saveCurrentLoadout(button.dataset.saveLoadout));
  });
  ui.panelContent.querySelectorAll("[data-activate-loadout]").forEach((button) => {
    button.addEventListener("click", () => activateSavedLoadout(button.dataset.activateLoadout));
  });
}

function saveCurrentLoadout(slot) {
  const label = `Build ${slot}`;
  if (game.online) {
    game.socket.emit(NET.PLAYER_SAVE_LOADOUT, { slot, label }, (ack) => {
      if (!ack?.ok) return toast("Could not save that loadout.");
      applyServerPlayerUpdate(ack.player);
      saveGame();
      toast(`Saved Loadout ${slot}.`);
      renderLoadoutPanel();
    });
    return;
  }
  const result = saveLoadout(game.save, slot, label);
  if (!result.ok) return toast("Could not save that loadout.");
  game.save = result.player;
  saveGame();
  toast(`Saved Loadout ${slot}.`);
  renderLoadoutPanel();
}

function activateSavedLoadout(slot) {
  if (game.online) {
    game.socket.emit(NET.PLAYER_ACTIVATE_LOADOUT, { slot }, (ack) => {
      if (!ack?.ok) return toast(loadoutReason(ack?.reason));
      applyServerPlayerUpdate(ack.player);
      applyVisualEquipment(game.playerGroup, game.save);
      saveGame();
      toast(`Activated Loadout ${slot}.`);
      renderLoadoutPanel();
    });
    return;
  }
  const result = activateLoadout(game.save, slot);
  if (!result.ok) return toast(loadoutReason(result.reason));
  game.save = result.player;
  applyVisualEquipment(game.playerGroup, game.save);
  saveGame();
  toast(`Activated Loadout ${slot}.`);
  renderLoadoutPanel();
}

function loadoutReason(reason) {
  if (reason === "ability") return "That loadout uses an unlearned ability.";
  if (reason === "item") return "That loadout uses missing equipment.";
  if (reason === "combat") return "Leave combat before changing loadouts.";
  return "That loadout is unavailable.";
}

function applyServerPlayerUpdate(player) {
  if (!player) return;
  Object.assign(game.save, {
    health: player.health ?? game.save.health,
    maxHealth: player.maxHealth ?? game.save.maxHealth,
    mana: player.mana ?? game.save.mana,
    maxMana: player.maxMana ?? game.save.maxMana,
    level: player.level ?? game.save.level,
    xp: player.xp ?? game.save.xp,
    coins: player.coins ?? game.save.coins,
    equipment: player.equipment || game.save.equipment,
    equippedWeapon: player.equippedWeapon || game.save.equippedWeapon,
    equippedArmor: player.equippedArmor || game.save.equippedArmor,
    magicPower: player.magicPower ?? game.save.magicPower,
    spentAttributes: player.spentAttributes || game.save.spentAttributes,
    availableAttributePoints: player.availableAttributePoints ?? game.save.availableAttributePoints,
    availableSkillPoints: player.availableSkillPoints ?? game.save.availableSkillPoints,
    inventory: player.inventory || game.save.inventory,
    learnedAbilities: player.learnedAbilities || game.save.learnedAbilities,
    hotbar: player.hotbar || game.save.hotbar,
    skillTreeNodes: player.skillTreeNodes || game.save.skillTreeNodes,
    savedBuilds: player.savedBuilds || game.save.savedBuilds,
    activeLoadout: player.activeLoadout || game.save.activeLoadout,
    waypoints: player.waypoints || game.save.waypoints,
    openedChests: player.openedChests || game.save.openedChests,
    bestiaryProgress: player.bestiaryProgress || game.save.bestiaryProgress,
    zoneCompletion: player.zoneCompletion || game.save.zoneCompletion,
    achievements: player.achievements || game.save.achievements,
    firstClearRewards: player.firstClearRewards || game.save.firstClearRewards,
    title: player.title || game.save.title
  });
  game.save = applyEquipment(game.save);
  if (game.playerGroup) applyVisualEquipment(game.playerGroup, game.save);
}

function allocateAttributePoint(attributeId) {
  if (isLocallyDead()) return toast("You cannot allocate stats while defeated.");
  if (game.online) {
    game.socket.emit(NET.PLAYER_SPEND_ATTRIBUTE, { attributeId, count: 1 }, (ack) => {
      if (!ack?.ok) return toast(ack?.reason === "points" ? "No attribute points available." : "That stat cannot be increased.");
      applyServerPlayerUpdate(ack.player);
      saveGame();
      playUiSound("tick");
      renderCharacterPanel();
    });
    return;
  }
  const result = spendAttributePoint(game.save, attributeId, 1);
  if (!result.ok) return toast(result.reason === "points" ? "No attribute points available." : "That stat cannot be increased.");
  game.save = result.player;
  saveGame();
  playUiSound("tick");
  renderCharacterPanel();
}

function useRestStoneItem() {
  if (isLocallyDead()) return toast("You cannot reset stats while defeated.");
  if (game.online) {
    game.socket.emit(NET.PLAYER_USE_REST_STONE, {}, (ack) => {
      if (!ack?.ok) return toast("No Rest Stone available.");
      applyServerPlayerUpdate(ack.player);
      saveGame();
      playUiSound("equip");
      toast("Rest Stone restored your points.");
      renderCharacterPanel();
    });
    return;
  }
  const result = useRestStone(game.save);
  if (!result.ok) return toast("No Rest Stone available.");
  game.save = result.player;
  saveGame();
  playUiSound("equip");
  toast("Rest Stone restored your points.");
  renderCharacterPanel();
}

function renderPartyPanel() {
  ui.panelTitle.textContent = "Party";
  ui.panelContent.innerHTML = `
    <p class="intro">Create a code and send it to friends. Parties support 2-4 players and share boss credit when members fight in the arena.</p>
    <button class="primary" id="create-party">Create Party Code</button>
    <div class="setup-grid">
      <input id="party-code" maxlength="4" placeholder="Code" />
      <button id="join-party">Join</button>
    </div>
    <button id="leave-party">Leave Party</button>
  `;
  ui.panelContent.querySelector("#create-party").addEventListener("click", () => {
    if (!game.online) {
      toast("Party codes need the online server.");
      return;
    }
    game.socket.emit(NET.PARTY_CREATE, {}, (ack) => {
      if (ack?.ok) toast(`Party code: ${ack.party.code}`);
    });
  });
  ui.panelContent.querySelector("#join-party").addEventListener("click", () => {
    if (!game.online) return toast("Party codes need the online server.");
    const code = ui.panelContent.querySelector("#party-code").value;
    game.socket.emit(NET.PARTY_JOIN, { code }, (ack) => {
      toast(ack?.ok ? `Joined party ${ack.party.code}.` : ack?.error || "Could not join party.");
    });
  });
  ui.panelContent.querySelector("#leave-party").addEventListener("click", () => {
    if (game.online) game.socket.emit(NET.PARTY_LEAVE, {}, () => toast("Left party."));
  });
}
