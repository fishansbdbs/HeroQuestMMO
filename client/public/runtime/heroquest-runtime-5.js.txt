  game.loot.delete(lootId);
  const mesh = game.lootMeshes.get(lootId);
  if (mesh) {
    game.scene.remove(mesh);
    game.lootMeshes.delete(lootId);
  }
  toast("Loot collected.");
}

function claimHealingOrb(orbId) {
  if (isLocallyDead()) {
    toast("You cannot absorb healing while defeated.");
    return;
  }
  const orb = game.healingOrbs.get(orbId);
  if (!orb) return toast("That healing orb has faded.");
  if (game.online) {
    game.socket.emit(NET.HEALING_ORB_CLAIM, { orbId }, (ack) => {
      if (!ack?.ok) return toast(healingOrbClaimReason(ack?.reason));
      if (ack.player) applyServerPlayerUpdate(ack.player);
      for (const heal of ack.heals || []) {
        if (heal.playerId === game.socket?.id) floatingText(`+${heal.amount}`, groupPosition(game.playerGroup), "#6df08a");
      }
      game.healingOrbs.delete(orbId);
      toast("Healing Orb absorbed.");
    });
    return;
  }
  toast("Healing Orb needs the online combat server.");
}

function applyLootBag(bag) {
  const inventoryResult = applyRewardItemsToInventory(game.save.inventory, bag.items || []);
  if (!inventoryResult.ok) {
    toast(inventoryResult.reason === "full" ? "Inventory full. Make space before collecting loot." : "That loot cannot be collected.");
    return false;
  }
  game.save.inventory = inventoryResult.inventory;
  game.save.coins += bag.coins || 0;
  if (bag.coins) toast(`Collected ${bag.coins} coins.`);
  for (const item of bag.items || []) {
    const found = getItem(item.itemId);
    toast(`${["Rare", "Epic"].includes(found?.rarity) ? "Rare drop: " : "Found "}${found?.name || item.itemId}.`);
  }
  playUiSound("pickup");
  saveGame();
  return true;
}

function applyRewardItemsToInventory(inventory, items = []) {
  let nextInventory = inventory || [];
  for (const item of items) {
    const result = addInventoryStack(nextInventory, item.itemId, item.quantity || 1);
    if (!result.ok) return result;
    nextInventory = result.inventory;
  }
  return { ok: true, inventory: nextInventory, overflow: [] };
}

function hitBoss(damage, kind = "auto") {
  if (isLocallyDead() || !game.boss || game.boss.defeated) return;
  game.boss.health = Math.max(0, game.boss.health - damage);
  game.boss.phase = game.boss.health <= game.boss.maxHealth * 0.5 ? 2 : 1;

function handleCombatAck(result) {
  if (!result?.ok) return;
  if (result.projectile) projectileEffect(result.projectile);
  if (result.areaEffect) areaEffect(result.areaEffect);
  if (result.meleeEffect) meleeEffect(result.meleeEffect);
  if (result.healingOrb) healingOrbEffect(result.healingOrb);
  if (result.healingEffect) healingBeamEffect(result.healingEffect);
  if (result.damage) {
    const target = result.result?.enemy || getCurrentTarget();
    if (target?.position) {
      floatingText(`-${result.damage}`, target.position, result.kind === "slash" ? "#ffe38b" : "#ffdd9a");
      hitSpark(target.position, result.kind);
      flashMesh(target.id === "shadow_wyrm" ? game.bossGroup : target.id === ICE_MAGE_BOSS.id ? game.iceMageGroup : game.enemyMeshes.get(target.id), 0xfff0a0);
      logCombat(`${result.kind === "slash" ? "Slash" : "Hit"} ${targetDisplayName(target)} for ${result.damage}.`);
    }
  }
  if (result.result?.defeated) {
    awardEnemy(result.result.enemyDef);
    if (result.result.lootBag) game.loot.set(result.result.lootBag.id, result.result.lootBag);
    if (result.result.enemy?.id) startEnemyDeath(result.result.enemy.id, result.result.enemyDef, result.result.enemy.position);
  }
  const bossResult = result.boss || result.result?.boss;
  if (bossResult?.boss) {
    game.boss = { ...game.boss, ...bossResult.boss };
  }
  if (result.boss && result.damage) {
    floatingText(`-${result.damage}`, { x: 0, y: 4, z: -1 }, "#ff8d76");
    hitSpark({ x: 0, y: 3.4, z: -2.2 }, result.kind);
    flashMesh(game.bossGroup, 0xff6b57);
  }
  if (bossResult?.defeated) {
    completeBossQuestLocal(bossResult.reward || { xp: BOSS.xp, coins: BOSS.coins });
    if (bossResult.lootBag) game.loot.set(bossResult.lootBag.id, bossResult.lootBag);
    toast("Shadow Wyrm defeated!");
  }
  const iceMageResult = result.iceMage || result.result?.iceMage;
  const iceMageSnapshot = iceMageResult?.iceMage || iceMageResult?.boss;
  if (iceMageSnapshot) {
    game.iceMage = { ...game.iceMage, ...iceMageSnapshot };
  }
  if (result.iceMage && result.damage) {
    floatingText(`-${result.damage}`, { x: 0, y: 4.8, z: -0.6 }, "#9ae8ff");
    hitSpark({ x: 0, y: 3.8, z: -0.8 }, result.kind);
    flashMesh(game.iceMageGroup, 0x8df2ff);
  }
  if (iceMageResult?.defeated) {
    completeIceMageQuestLocal(iceMageResult.reward || { xp: ICE_MAGE_BOSS.xp, coins: ICE_MAGE_BOSS.coins });
    if (iceMageResult.lootBag) game.loot.set(iceMageResult.lootBag.id, iceMageResult.lootBag);
    toast("Zero, the Ice Mage defeated!");
  }
}

function damagePlayer(amount) {
  if (isLocallyDead()) return;
  markCombat();
  const guarded = performance.now() < game.guardUntil;
  const finalAmount = guarded ? Math.max(1, Math.round(amount * (1 - ABILITIES.guard.damageReduction))) : amount;
  game.save.health = Math.max(0, game.save.health - finalAmount);
  floatingText(guarded ? `Guard -${finalAmount}` : `-${finalAmount}`, groupPosition(game.playerGroup), guarded ? "#9ad5ff" : "#ff8080");
  flashMesh(game.playerGroup, guarded ? 0x9ad5ff : 0xff6b57);
  logCombat(guarded ? `Guard reduced ${amount} to ${finalAmount}.` : `Took ${finalAmount} damage.`);
  if (!guarded) screenShake(0.18);
  playUiSound("hurt");
  if (game.save.health <= 0) {
    beginDefeat();
  }
  saveGame();
}

function isLocallyDead() {
  return game.save.state === "dead" || (game.save.health ?? 0) <= 0;
}

function beginDefeat(respawnAt = Date.now() + 5000) {
  if (game.save.state === "dead" && game.respawnAt) return;
  game.save.state = "dead";
  game.save.health = 0;
  game.defeatedAt = performance.now();
  game.respawnAt = respawnAt;
  game.respawnRequested = false;
  game.autoAttack = false;
  selectTarget(null);
  clearTelegraphs();
  closePanel();
  toast("You were defeated.");
  logCombat("Defeated. Respawning at Dawnrest soon.");
  saveGame();
}

function updateDefeatState() {
  const dead = isLocallyDead();
  ui.defeatOverlay?.classList.toggle("hidden", !dead);
  if (!dead) return;
  const remaining = Math.max(0, Math.ceil(((game.respawnAt || Date.now()) - Date.now()) / 1000));
  if (ui.defeatTimer) ui.defeatTimer.textContent = remaining > 0 ? `Respawning in ${remaining}s` : "Respawn ready";
  if (ui.respawnButton) ui.respawnButton.disabled = remaining > 0;
}

function requestRespawn() {
  if (!isLocallyDead() || game.respawnRequested) return;
  if (Date.now() < (game.respawnAt || 0)) return;
  game.respawnRequested = true;
  if (game.online) {
    game.socket.emit(NET.PLAYER_RESPAWN, {}, (ack) => {
      if (ack?.ok) finishRespawn(ack.player);
      else {
        game.respawnRequested = false;
        toast(respawnReason(ack?.reason));
        logCombat(respawnReason(ack?.reason));
      }
    });
    return;
  }
  finishRespawn();
}

function finishRespawn(serverPlayer = null) {
  game.save.state = "alive";
  game.save.health = serverPlayer?.health || game.save.maxHealth;
  game.save.mana = serverPlayer?.mana ?? game.save.maxMana;
  game.defeatedAt = null;
  game.respawnAt = null;
  game.respawnRequested = false;
  ui.defeatOverlay?.classList.add("hidden");
  loadZone(ZONES.HUB, true);
  game.save.health = game.save.maxHealth;
  game.save.mana = serverPlayer?.mana ?? game.save.maxMana;
  saveGame();
  toast("Respawned at Dawnrest.");
}

function respawnReason(reason) {
  if (reason === "not_ready") return "Respawn is not ready yet.";
  if (reason === "not_dead") return "You are already alive.";
  return "Respawn failed. Try again.";
}

function spawnLocalEnemies() {
  game.enemies.clear();
  const spawns = game.zone === ZONES.FROSTVEIL ? FROSTVEIL_SPAWNS : FIELD_SPAWNS;
  for (const spawn of spawns) {
    for (let i = 0; i < spawn.count; i += 1) {
      const angle = Math.random() * Math.PI * 2;
      const radius = Math.random() * spawn.radius;
      const def = getEnemy(spawn.enemyId);
      const id = `local_enemy_${game.localEnemyId++}`;
      game.enemies.set(id, {
        id,
        enemyId: spawn.enemyId,
        health: def.maxHealth,
        maxHealth: def.maxHealth,
        position: {
          x: spawn.center[0] + Math.cos(angle) * radius,
          y: 0,
          z: spawn.center[2] + Math.sin(angle) * radius
        }
      });
    }
  }
}

function resolveBossAttack(attackOrType) {
  const attack = typeof attackOrType === "string" ? createClientBossAttack(attackOrType) : attackOrType;
  const type = attack.type;
  bossImpactEffect(attack);
  if (type === "shadow_summon") {
    for (let i = 0; i < 3; i += 1) {
      const def = ENEMIES.shadow_slime;
      const id = `shadow_${game.localEnemyId++}`;
      const point = attack.shape?.points?.[i];
      const a = Math.random() * Math.PI * 2;
      game.enemies.set(id, {
        id,
        enemyId: def.id,
        health: def.maxHealth,
        maxHealth: def.maxHealth,
        position: point ? { x: point.x, y: 0, z: point.z } : { x: Math.cos(a) * 8, y: 0, z: Math.sin(a) * 8 }
      });
    }
    toast("The Shadow Wyrm summons slimes.");
    return;
  }
  const pos = groupPosition(game.playerGroup);
  const dist = distance2d(pos, { x: 0, z: 0 });
  const hit = type === "fire_cone"
    ? pos.z > -7 && Math.abs(pos.x) < 10 && dist < 20
    : type === "tail_sweep"
      ? dist >= 6.5 && dist <= 13.5
      : type === "shadow_slam"
        ? distance2d(pos, attack.shape) <= attack.shape.radius
        : dist < 15;
  const scale = type === "wing_gust" ? 0.62 : type === "tail_sweep" ? 0.95 : type === "shadow_slam" ? 1.15 : 1.05;
  if (hit) damagePlayer(calculateIncomingDamage(Math.round(BOSS.damage * scale), game.save));
  if (type === "wing_gust") {
    const push = new THREE.Vector3(pos.x, 0, pos.z).normalize().multiplyScalar(4);
    game.playerGroup.position.add(push);
  }
  if (type === "tail_sweep") {
    const push = new THREE.Vector3(pos.x, 0, pos.z).normalize().multiplyScalar(3.2);
    game.playerGroup.position.add(push);
  }
}

function resolveIceMageAttack(attackOrType) {
  const attack = typeof attackOrType === "string" ? createClientIceMageAttack(attackOrType) : attackOrType;
  bossImpactEffect(attack);
  if (attack.type === "frozen_servants") {
    for (const [index, point] of (attack.shape?.points || []).entries()) {
      const enemyId = index % 2 === 0 ? "frost_wisp" : "frozen_knight";
      const def = getEnemy(enemyId);
      const id = `ice_servant_${game.localEnemyId++}`;
      game.enemies.set(id, {
        id,
        enemyId,
        zone: ZONES.PALACE,
        health: def.maxHealth,
        maxHealth: def.maxHealth,
        position: { x: point.x, y: 0, z: point.z }
      });
    }
    toast("Frozen servants answer Zero's call.");
    return;
  }
  if (attack.type === "ice_wall") {
    toast("Ice walls fracture and fade.");
    return;
  }
  if (playerInIceMageAttack(attack)) {
    const damage = calculateIncomingDamage(Math.round(ICE_MAGE_BOSS.damage * (attack.damageScale || 1)), game.save);
    damagePlayer(damage);
  }
}

function playerInIceMageAttack(attack) {
  const pos = groupPosition(game.playerGroup);
  const shape = attack.shape || {};
  if (shape.kind === "line") return Math.abs(pos.x - (shape.x || 0)) <= (shape.width || 3) / 2 && pos.z >= -16 && pos.z <= 20;
  if (shape.kind === "circle") return distance2d(pos, shape) <= shape.radius;
  if (shape.kind === "ring") {
    const dist = distance2d(pos, shape);
    return dist >= shape.innerRadius && dist <= shape.outerRadius;
  }
  if (shape.kind === "sequence") {
    return (shape.circles || []).some((circle) => distance2d(pos, circle) <= circle.radius) ||
      (shape.lines || []).some((line) => Math.abs(pos.x - line.x) <= line.width / 2 && pos.z >= line.zMin && pos.z <= line.zMax);
  }
  return false;
}

function showTelegraph(attackOrType) {
  clearTelegraphs();
  const attack = typeof attackOrType === "string" ? createClientBossAttack(attackOrType) : attackOrType;
  const type = attack.type;
  const group = new THREE.Group();
  if (type === "fire_cone") {
    const shape = new THREE.Shape();
    shape.moveTo(0, 0);
    shape.lineTo(-8, 17);
    shape.lineTo(8, 17);
    shape.lineTo(0, 0);
    const mesh = new THREE.Mesh(new THREE.ShapeGeometry(shape), mat(0xff5533, 0.36, true));
    mesh.rotation.x = -Math.PI / 2;
    mesh.position.set(0, 0.12, -4);
    group.add(mesh);
  } else if (type === "tail_sweep") {
    const mesh = new THREE.Mesh(new THREE.RingGeometry(7, 13, 64, 2, Math.PI * 0.18, Math.PI * 1.45), mat(0xff8b3d, 0.36, true));
    mesh.rotation.x = -Math.PI / 2;
    mesh.position.y = 0.13;
    group.add(mesh);
  } else if (attack.shape?.kind === "summon") {
    for (const point of attack.shape?.points || []) {
      const mesh = new THREE.Mesh(new THREE.CircleGeometry(attack.shape.radius || 3.2, 40), mat(0x9a55ff, 0.34, true));
      mesh.rotation.x = -Math.PI / 2;
      mesh.position.set(point.x, 0.13, point.z);
      group.add(mesh);
    }
  } else if (attack.shape?.kind === "line") {
    const mesh = new THREE.Mesh(new THREE.BoxGeometry(attack.shape.width || 3, 0.05, attack.shape.length || 28), mat(0x49d8ff, 0.34, true));
    mesh.position.set(attack.shape.x || 0, 0.14, attack.shape.z || 0);
    group.add(mesh);
  } else if (attack.shape?.kind === "ring") {
    const mesh = new THREE.Mesh(new THREE.RingGeometry(attack.shape.innerRadius || 6, attack.shape.outerRadius || 15, 72), mat(0x62dfff, 0.34, true));
    mesh.rotation.x = -Math.PI / 2;
    mesh.position.set(attack.shape.x || 0, 0.14, attack.shape.z || 0);
    group.add(mesh);
  } else if (attack.shape?.kind === "walls") {
    for (const wall of attack.shape.walls || []) {
      const mesh = new THREE.Mesh(new THREE.BoxGeometry(wall.width || 2, 0.08, wall.length || 14), mat(0x8eeaff, 0.38, true));
      mesh.position.set(wall.x || 0, 0.16, wall.z || 0);
      group.add(mesh);
    }
  } else if (attack.shape?.kind === "sequence") {
    for (const line of attack.shape.lines || []) {
      const length = Math.abs((line.zMax || 0) - (line.zMin || 0));
      const mesh = new THREE.Mesh(new THREE.BoxGeometry(line.width || 3, 0.05, length || 28), mat(0x59dfff, 0.34, true));
      mesh.position.set(line.x || 0, 0.14, ((line.zMin || 0) + (line.zMax || 0)) / 2);
      group.add(mesh);
    }
    for (const circle of attack.shape.circles || []) {
      const mesh = new THREE.Mesh(new THREE.CircleGeometry(circle.radius || 3.2, 40), mat(0x59dfff, 0.32, true));
      mesh.rotation.x = -Math.PI / 2;
      mesh.position.set(circle.x || 0, 0.14, circle.z || 0);
      group.add(mesh);
    }
  } else {
    const radius = attack.shape?.radius || (type === "wing_gust" ? 15 : 4.8);
    const color = type.startsWith("ice") || type.startsWith("frost") || type.startsWith("blizzard") || type === "shatter" ? 0x59dfff : type === "shadow_slam" ? 0xff3d3d : 0xff7d3d;
    const mesh = new THREE.Mesh(new THREE.CircleGeometry(radius, 56), mat(color, 0.29, true));
    mesh.rotation.x = -Math.PI / 2;
    mesh.position.set(attack.shape?.x || 0, 0.13, attack.shape?.z || 0);
    group.add(mesh);
  }
  const label = makeLabel(attack.name || bossAttackText(type), attack.shape?.kind === "summon" ? "#e8d4ff" : type.startsWith("ice") || type.startsWith("frost") || type.startsWith("blizzard") || type === "zero_hour" || type === "shatter" ? "#d8fbff" : "#ffd7b0");
  const labelPos = attack.shape?.kind === "circle" ? attack.shape : { x: 0, z: -2 };
  label.position.set(labelPos.x || 0, 2.4, labelPos.z || 0);
  group.add(label);
  game.scene.add(group);
  animateTelegraph(group, attack);
  game.telegraphs.push(group);
}

function animateTelegraph(group, attack) {
  const start = performance.now();
  const delay = Math.max(400, (attack.resolvesAt || start + 1100) - (attack.startedAt || start));
  const animate = () => {
    const t = Math.min(1, (performance.now() - start) / delay);
    group.scale.setScalar(1 + Math.sin(t * Math.PI) * 0.035);
    group.traverse((child) => {
      if (child.isMesh && child.material) child.material.opacity = Math.max(0.12, 0.38 - t * 0.12);
    });
    if (t < 1 && game.telegraphs.includes(group)) requestAnimationFrame(animate);
  };
  animate();
}

function bossImpactEffect(attack) {
  const center = attack.shape?.kind === "circle" || attack.shape?.kind === "ring" ? attack.shape : { x: 0, z: attack.type === "fire_cone" ? -7 : 0 };
  const icy = attack.type?.startsWith("ice") || attack.type?.startsWith("frost") || attack.type?.startsWith("blizzard") || attack.type === "zero_hour" || attack.type === "shatter";
  const color = attack.shape?.kind === "summon" ? 0x9a55ff : icy ? 0x65dfff : attack.type === "wing_gust" ? 0xffd08a : 0xff5d3d;
  const radius = attack.shape?.outerRadius || attack.shape?.radius || (attack.shape?.kind === "line" ? 7 : 4.5);
  const ring = new THREE.Mesh(new THREE.RingGeometry(0.8, radius, 48), mat(color, 0.36, true));
  ring.rotation.x = -Math.PI / 2;
  ring.position.set(center.x || 0, 0.18, center.z || 0);
  game.scene.add(ring);
  screenShake(attack.type === "wing_gust" ? 0.08 : 0.16);
  const started = performance.now();
  const animate = () => {
    const t = (performance.now() - started) / 420;
    ring.scale.setScalar(1 + t * 0.5);
    ring.material.opacity = Math.max(0, 0.36 - t * 0.36);
    if (t < 1) requestAnimationFrame(animate);
    else game.scene.remove(ring);
  };
  animate();
}

function clearTelegraphs() {
  game.telegraphs.forEach((mesh) => game.scene.remove(mesh));
  game.telegraphs = [];
}

function applySnapshot(snapshot) {
  if (!snapshot || snapshot.zone !== game.zone) return;
  if (snapshot.serverTime) game.lastPingMs = Math.max(0, Math.round(Date.now() - snapshot.serverTime));
  if (snapshot.enemies) {
    game.enemies.clear();
    snapshot.enemies.forEach((enemy) => game.enemies.set(enemy.id, enemy));
    if (game.zone === ZONES.HUB) ensureTrainingDummy();
  }
  if (game.targetId && game.targetId !== "shadow_wyrm" && game.targetId !== ICE_MAGE_BOSS.id && !game.enemies.has(game.targetId)) selectTarget(null);
  if (snapshot.loot) {
    game.loot.clear();
    snapshot.loot.forEach((bag) => game.loot.set(bag.id, bag));
  }
  if (snapshot.healingOrbs) {
    game.healingOrbs.clear();
    snapshot.healingOrbs.forEach((orb) => game.healingOrbs.set(orb.id, orb));
  }
  if (snapshot.boss) {
    game.boss = { ...game.boss, ...snapshot.boss };
  }
  if (snapshot.iceMage) {
    game.iceMage = { ...game.iceMage, ...snapshot.iceMage };
  }
  if (snapshot.publicEvent) {
    game.publicEvent = snapshot.publicEvent;
  }
